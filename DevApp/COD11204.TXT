OBJECT Codeunit 11204 Inward Reg.-Reverse (Yes/No)
{
  OBJECT-PROPERTIES
  {
    Date=;
    Time=;
    Version List=NAVSE7.00;
  }
  PROPERTIES
  {
    TableNo=11200;
    OnRun=BEGIN
            InwHeader.COPY(Rec);
            Code;
            Rec := InwHeader;
          END;

  }
  CODE
  {
    VAR
      ReverseInward@1070000 : Codeunit 11203;
      InwHeader@1070001 : Record 11200;
      Text001@1070002 : TextConst 'ENU=Do you want to reverse the Inward Registration?';
      Text002@1070003 : TextConst 'ENU=In order to reverse, the status has to be Posted.';

    PROCEDURE Code@1070001();
    BEGIN
      IF InwHeader.Status = InwHeader.Status::Posted THEN BEGIN
        IF CONFIRM(Text001,FALSE) THEN
          ReverseInward.RUN(InwHeader);
      END ELSE
        ERROR(Text002);
    END;

    BEGIN
    END.
  }
}
