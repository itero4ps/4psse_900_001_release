    PROCEDURE SetFormFilter@1100285001();
    VAR
      exUserComp@1100285000 : Record 12013641;
    BEGIN
      RESET;
      SETCURRENTKEY("Company Name","Document Type","Document No.");

      FILTERGROUP := 2;
      SETFILTER("Document Type",'%1|%2',"Document Type"::"Posted Invoice","Document Type"::"Posted Credit Memo");

      exUserComp.SETRANGE("Company Name", COMPANYNAME);
      exUserComp.SETRANGE("User ID", USERID);
      exUserComp.SETRANGE("All-Company Access to Ex Docs", TRUE);
      IF NOT exUserComp.FINDFIRST THEN
        SETRANGE("Company Name", COMPANYNAME);
      FILTERGROUP := 0;
    END;
