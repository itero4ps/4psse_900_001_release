    PROCEDURE EvaluateVendorCategory@1100528901();
    VAR
      VendorRec@1100528901 : Record 23;
      VendorCategoryRec@1100528902 : Record 11071892;
      ActivateMark@1100528903 : Boolean;
    BEGIN
      //**4PS
      CLEARMARKS;
      MARKEDONLY(FALSE);
      ActivateCategoryFilter := FALSE;

      IF CodingSystem + VendorCategoryFilter + VendorQualificationCodeFilter = '' THEN
        EXIT;

      VendorRec.COPY(Rec);
      IF VendorRec.FINDSET THEN BEGIN
        REPEAT
          VendorCategoryRec.SETRANGE("Vendor No.", VendorRec."No.");
          VendorCategoryRec.SETRANGE("Coding System", CodingSystem);
          VendorCategoryRec.SETFILTER(Code, VendorCategoryFilter);
          VendorCategoryRec.SETFILTER("Vendor Qualification Code", VendorQualificationCodeFilter);
          IF VendorCategoryRec.FINDFIRST THEN BEGIN
            ActivateCategoryFilter := TRUE;
            VendorRec.MARK := TRUE;
          END;
        UNTIL VendorRec.NEXT = 0;
      END;

      VendorRec.MARKEDONLY(TRUE);
      Rec.COPY(VendorRec);
    END;
