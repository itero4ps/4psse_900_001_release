    PROCEDURE CheckICInterestAllSelectOfAcc@1100485001();
    VAR
      lvRec1@1100485001 : Record 81;
      lvRec2@1100485000 : Record 81;
      lvInterestLineICAccRec@1100485002 : Record 11020307;
      lvInterestLineICAccRec2@1100485003 : Record 11020307;
      lvText11012001@1100485004 : TextConst 'ENU=Not all interest lines (f.i. line ''%1'') of IC Account ''%2'' are selected, all interest lines of an IC-Account must be posted simultaneously.';
    BEGIN
      //**4PS
      //* All interest lines of an IC account should be posted at once.

      lvRec1.COPY(Rec);
      lvRec2.COPY(Rec);

      WITH lvInterestLineICAccRec DO BEGIN
        RESET;
        SETCURRENTKEY(Type,"Gen. Journal Template Name","Gen. Journal Batch Name","Gen. Journal Line No.");
        SETRANGE(Type, Type::GenJnlLine);
        SETRANGE("Gen. Journal Template Name", "Journal Template Name");
        SETRANGE("Gen. Journal Batch Name", "Journal Batch Name");
        IF NOT FINDFIRST THEN
          EXIT;

        IF lvRec1.FINDSET THEN BEGIN
          REPEAT
            SETRANGE("Gen. Journal Line No.", lvRec1."Line No.");
            IF FINDFIRST THEN BEGIN
              lvInterestLineICAccRec2.RESET;
              lvInterestLineICAccRec2.SETRANGE(Type, Type::GenJnlLine);
              lvInterestLineICAccRec2.SETRANGE("IC Account No.", "IC Account No.");
              lvInterestLineICAccRec2.SETRANGE("Gen. Journal Template Name", "Gen. Journal Template Name");
              lvInterestLineICAccRec2.SETRANGE("Gen. Journal Batch Name", "Gen. Journal Batch Name");
              lvInterestLineICAccRec2.SETFILTER("Gen. Journal Line No.", '<>%1', "Gen. Journal Line No.");
              IF lvInterestLineICAccRec2.FINDSET THEN BEGIN
                REPEAT
                  lvRec2.SETRANGE("Journal Template Name", "Gen. Journal Template Name");
                  lvRec2.SETRANGE("Journal Batch Name", "Gen. Journal Batch Name");
                  lvRec2.SETRANGE("Line No.", lvInterestLineICAccRec2."Gen. Journal Line No.");
                  IF NOT lvRec2.FINDFIRST THEN    //* Do not use 'GET', should be tested on filters.
                    ERROR(lvText11012001, lvInterestLineICAccRec2."Gen. Journal Line No.", "IC Account No.");
                UNTIL lvInterestLineICAccRec2.NEXT = 0;
              END;
            END;
          UNTIL lvRec1.NEXT = 0;
        END;
      END;
    END;
