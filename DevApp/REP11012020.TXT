    LOCAL PROCEDURE FillTempErrorTable@1100528400(VAR Vendor2@1100528402 : Record 23;VAR TempError@1100528403 : TEMPORARY Record 11012051;DueDate2@1100528405 : Date;IgnoreDueDateOlderThan2@1100528404 : Date);
    VAR
      WKACheck@1100528401 : Record 11020275;
      VendorWKACheck@1100528400 : Record 11020276;
    BEGIN
      TempError."Line No." := 0;

      IF Vendor2."Due Date COC Statement" < DueDate2 THEN
        CreateError(TempError, Vendor2, Text001);

      WKACheck.SETRANGE(Request, TRUE);
      WKACheck.SetWKAVendorFilters(Vendor2);
      IF WKACheck.FINDSET THEN
        REPEAT
          IF NOT VendorWKACheck.GET(Vendor2."No.", WKACheck.Code) THEN
            CreateError(TempError, Vendor2, WKACheck.Description)
          ELSE BEGIN
            VendorWKACheck.SetIgnoreDueDateOlderThan(IgnoreDueDateOlderThan2);
            IF NOT VendorWKACheck.DataIsValid(DueDate2) THEN
              CreateError(TempError, Vendor2, WKACheck.Description);
          END;
        UNTIL (WKACheck.NEXT = 0);

      VendorWKACheck.SETRANGE("Vendor No.", Vendor2."No.");
      VendorWKACheck.SETRANGE(Mandatory, FALSE);
      VendorWKACheck.SETRANGE("Extra Request Vendor", TRUE);
      IF VendorWKACheck.FINDSET THEN
        REPEAT
          VendorWKACheck.SetIgnoreDueDateOlderThan(IgnoreDueDateOlderThan2);
          IF NOT VendorWKACheck.DataIsValid(DueDate2) THEN BEGIN
            VendorWKACheck.CALCFIELDS(Description);
            CreateError(TempError, Vendor2, VendorWKACheck.Description);
          END;
        UNTIL (VendorWKACheck.NEXT = 0);
    END;

    BEGIN
    END.
