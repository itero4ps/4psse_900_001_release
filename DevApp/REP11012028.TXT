    PROCEDURE GetReportName@1100525003() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));
    END;

    PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100525000 : Text;
      ProjectFilter2@1100525001 : Text;
    BEGIN
      ProjectFilter0 := Job.GETFILTERS;
      Job.FILTERGROUP(2);
      ProjectFilter2 := Job.GETFILTERS;
      Job.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Job.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Job.TABLECAPTION, ProjectFilter0));
    END;

    PROCEDURE CheckFilterReportSelection@1210190000();
    BEGIN
      //db, 21-06-05 (melding C-0004709)
      IF Job.GETFILTER("Period Filter") <> '' THEN
        PeriodFilter := Job.GETFILTER("Period Filter");
    END;

    PROCEDURE RemoveFlowFieldFilters@1100485002();
    VAR
      FieldRec@1100485004 : Record 2000000041;
      RecRef@1100485001 : RecordRef;
      FldRef@1100485003 : FieldRef;
      i@1100485002 : Integer;
    BEGIN
      RecRef.GETTABLE(Job);
      FOR i := 1 TO RecRef.FIELDCOUNT DO BEGIN
        FldRef := RecRef.FIELDINDEX(i);
        FieldRec.GET(RecRef.NUMBER, FldRef.NUMBER);
        IF FieldRec.Class = FieldRec.Class::FlowField THEN
          FldRef.SETRANGE;
      END;
      Job.SETVIEW(RecRef.GETVIEW);
    END;

    PROCEDURE CheckDataWithinProjFilters@1100485001() : Boolean;
    VAR
      FieldRec@1210190003 : Record 2000000041;
      RecRef@1210190002 : RecordRef;
      RecRef2@1210190004 : RecordRef;
      FldRef@1210190001 : FieldRef;
      FldRef2@1210190005 : FieldRef;
      i@1210190000 : Integer;
    BEGIN
      RecRef.GETTABLE(SaveProjectFilter);
      RecRef2.GETTABLE(Job);
      FOR i := 1 TO RecRef.FIELDCOUNT DO BEGIN
        FldRef := RecRef.FIELDINDEX(i);
        FldRef2 := RecRef2.FIELDINDEX(i);
        FieldRec.GET(RecRef.NUMBER, FldRef.NUMBER);
        IF FieldRec.Class = FieldRec.Class::FlowField THEN
          FldRef2.SETFILTER(FldRef.GETFILTER);
      END;

      EXIT(RecRef2.FIND('='));
    END;

    PROCEDURE PrintMainSubProject@1100525001() : Boolean;
    BEGIN
      IF (ProjectLevel = ProjectLevel::All) AND (Job."Single/Main/Sub Project" <> Job."Single/Main/Sub Project"::Single) THEN
        EXIT(TRUE);
    END;

    PROCEDURE PrintSingleProject@1100525002() : Boolean;
    BEGIN
      IF (Job."Single/Main/Sub Project" = Job."Single/Main/Sub Project"::Single) OR
         ((Job."Single/Main/Sub Project" = Job."Single/Main/Sub Project"::"Main Project") AND (ProjectLevel = ProjectLevel::Main)) THEN
        EXIT(TRUE);
    END;

    PROCEDURE PrintMainProjectGroupFooter@1100525004() : Boolean;
    BEGIN
      IF (ProjectLevel = ProjectLevel::All) AND (Job."Single/Main/Sub Project" <> Job."Single/Main/Sub Project"::Single) THEN
        EXIT(TRUE);
    END;

    BEGIN
    END.
