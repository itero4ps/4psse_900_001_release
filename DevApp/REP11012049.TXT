                                    ControlProjectCostObject.SETRANGE("Main Project No.", "Prognosis Line"."Main Project No.");
                                    IF Job."Single/Main/Sub Project" = Job."Single/Main/Sub Project"::"Main Project" THEN
                                      ControlProjectCostObject.SETRANGE("Project Filter")
                                    ELSE
                                      ControlProjectCostObject.SETRANGE("Project Filter", Job."No.");
                                    //

                                    Job.COPYFILTER("Progress Filter", ControlProjectCostObject."Progress Filter");
                                    ControlProjectCostObject.SETFILTER("Cumulation Filter", '..%1', "Prognosis Date");

                                    IF NOT ControlProjectCostObject.GET("Project No.", "Cost Type", "Cost Object") THEN
                                      ControlProjectCostObject.INIT
                                    ELSE
                                      ControlProjectCostObject.CALCFIELDS(Budget, "Actual Costs");
                                    BudgetAmount := ControlProjectCostObject.Budget;
                                    ActualCosts := ControlProjectCostObject."Actual Costs";
                                    TotalCost := CostControlManagement.TotCostCostObj(ControlProjectCostObject,TRUE);
                                    ControlProjectCostObject.CalculateAllowedOnly(AllowedCosts, AllowedHours);
                                  END;
                                  PrognosisResult := BudgetAmount - TotalCost - "Amount incl. Surcharge";
                                END;
