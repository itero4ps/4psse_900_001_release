    PROCEDURE TotalsGLPerProjType@1210190031(AccountNo@1210190000 : Code[20];CostType@1210190001 : 'Labor,Material,Subcontracting,Plant,Sundry,Revenues');
    VAR
      GLEntry@1100525001 : Record 17;
      Job@1100525000 : Record 11072003;
    BEGIN
      WITH GLEntry DO BEGIN
        SETCURRENTKEY("G/L Account No.","Posting Date");
        SETRANGE("G/L Account No.",AccountNo);
        SETRANGE("Posting Date", StartDate, EndDate);
        SETFILTER("Job No.", '<>%1', '');
        IF FINDSET THEN
          REPEAT
            IF Job.GET("Job No.") THEN
              IF (Job."Project Type" = ProjType.Code) AND (Job."Project Status" < Job."Project Status"::Finished) THEN
                CASE CostType OF
                  0: TotalGL[1] := TotalGL[1] + GLEntry.Amount;
                  1: TotalGL[2] := TotalGL[2] + GLEntry.Amount;
                  2: TotalGL[3] := TotalGL[3] + GLEntry.Amount;
                  3: TotalGL[4] := TotalGL[4] + GLEntry.Amount;
                  4: TotalGL[5] := TotalGL[5] + GLEntry.Amount;
                  5: TotalGL[6] := TotalGL[6] + GLEntry.Amount;
                END;
          UNTIL NEXT = 0;
      END;
    END;

    PROCEDURE TotalsPPPerProjType@1210190002(CostType@1210190000 : 'Labor,Material,Subcontracting,Plant,Sundry,Revenues');
    VAR
      JobLedgerEntry@1100525000 : Record 11072005;
      Job@1100525001 : Record 11072003;
    BEGIN
      WITH JobLedgerEntry DO BEGIN
        RESET;
        SETCURRENTKEY("Job No.","Cost Type","Entry Type","Posting Date");
        SETRANGE("Posting Date", StartDate, EndDate);
        SETFILTER("Job No.", '<>%1', '');
        SETRANGE("Project Type", ProjType.Code);
        IF CostType <> 5 THEN
          SETRANGE("Cost Type", CostType);
        IF FINDSET THEN
          REPEAT
            Job.GET("Job No.");
            IF Job."Project Status" < Job."Project Status"::Finished THEN
              CASE CostType OF
                0: TotalPP[1] := TotalPP[1] + "Total Cost (LCY)";
                1: TotalPP[2] := TotalPP[2] + "Total Cost (LCY)";
                2: TotalPP[3] := TotalPP[3] + "Total Cost (LCY)";
                3: TotalPP[4] := TotalPP[4] + "Total Cost (LCY)";
                4: TotalPP[5] := TotalPP[5] + "Total Cost (LCY)";
                5: TotalPP[6] := TotalPP[6] + "Total Price (LCY)";
              END;
          UNTIL JobLedgerEntry.NEXT = 0;
      END;
    END;
