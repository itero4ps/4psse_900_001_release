    LOCAL PROCEDURE BuildTempTables@1100485001();
    VAR
      lvIntercompanyRelationRec@1100485000 : Record 11012057;
      PostedHourLine@1100485001 : Record 11012085;
      lvTempCompanyRec@1100485002 : TEMPORARY Record 2000000006;
    BEGIN
      //**4PS.c; rewritten to new functionality (FO Msquared)

      lvTempCompanyRec.Name := COMPANYNAME;
      lvTempCompanyRec.INSERT;
      lvIntercompanyRelationRec.SETFILTER("Supplying Company", '<>%1', COMPANYNAME);
      lvIntercompanyRelationRec.SETRANGE("Receiving Company", COMPANYNAME);
      IF lvIntercompanyRelationRec.FINDSET THEN BEGIN
        REPEAT
          lvTempCompanyRec.Name := lvIntercompanyRelationRec."Supplying Company";
          lvTempCompanyRec.INSERT;
        UNTIL lvIntercompanyRelationRec.NEXT = 0;
      END;

      IF lvTempCompanyRec.FINDSET THEN
        REPEAT
          PostedHourLine.RESET;

          PostedHourLine.CHANGECOMPANY(lvTempCompanyRec.Name);
          IF "Posted Hour Line".GETFILTER("Service Order No.") <> '' THEN
            PostedHourLine.SETCURRENTKEY("Service Order No.");
          IF "Posted Hour Line".GETFILTER("Project No.") <> '' THEN
            PostedHourLine.SETCURRENTKEY("Project No.", Year, Week, "Employee No.");
          "Posted Hour Line".FILTERGROUP(2);
          PostedHourLine.COPYFILTERS("Posted Hour Line");
          "Posted Hour Line".FILTERGROUP(0);
          PostedHourLine.SETFILTER("Employee No.", '<>%1', '');
          PostedHourLine.SETFILTER("Supplying Company", '%1', '');
          PostedHourLine.SETRANGE("Component Type", PostedHourLine."Component Type"::Hours);
          PostedHourLine.SETRANGE(Type, HourType);
          IF HourType = HourType::Project THEN
            PostedHourLine.SETFILTER("Project No.", "Posted Hour Line".GETFILTER("Project No."));
          IF HourType = HourType::Service THEN
            PostedHourLine.SETFILTER("Service Order No.", "Posted Hour Line".GETFILTER("Service Order No."));
          IF PostedHourLine.FINDSET THEN
            REPEAT
              IF PostedHourLine."Receiving Company" <> '' THEN BEGIN
                Job.CHANGECOMPANY(PostedHourLine."Receiving Company");
                ServiceOrder.CHANGECOMPANY(PostedHourLine."Receiving Company");
              END ELSE BEGIN
                Job.CHANGECOMPANY(lvTempCompanyRec.Name);
                ServiceOrder.CHANGECOMPANY(lvTempCompanyRec.Name);
              END;
              IF NOT Job.GET(PostedHourLine."Project No.") THEN
                Job.INIT;
              Employee.CHANGECOMPANY(lvTempCompanyRec.Name);
              IF NOT Employee.GET(PostedHourLine."Employee No.") THEN
                Employee.INIT;
              IF NOT ServiceOrder.GET(PostedHourLine."Service Order No.") THEN
                ServiceOrder.INIT;
              IF PostedHourLine."Project No." <> '' THEN BEGIN
                JobType := FORMAT(HourType::Project);
                JobNo := PostedHourLine."Project No.";
              END;
              IF PostedHourLine."Service Order No." <> '' THEN BEGIN
                JobType := FORMAT(HourType::Service);
                JobNo := PostedHourLine."Service Order No.";
                Job."Print Manday Register" := TRUE;  //missing option for service
                IF CollectBaseOrder THEN BEGIN
                  IF ServiceOrder."Base Service Order No." <> '' THEN BEGIN
                    JobNo := ServiceOrder."Base Service Order No.";
                    IF NOT ServiceOrder.GET(JobNo) THEN
                      ServiceOrder.INIT;
                  END;
                END;
              END;
              IF (Job."Print Manday Register") AND (Employee."Print Manday Register") THEN BEGIN
                IF (NOT GenericTempTable.GET(JobNo,
                          FORMAT(PostedHourLine.Year, 4),
                          FORMAT(PostedHourLine.Week, 2),
                          '',
                          PostedHourLine."Employee No.",
                          JobType))
                THEN BEGIN
                  GenericTempTable.INIT;
                  GenericTempTable."Key Code1" := JobNo;
                  GenericTempTable."Key Code2" := FORMAT(PostedHourLine.Year, 4);
                  GenericTempTable."Key Code3" := FORMAT(PostedHourLine.Week, 2);
                  GenericTempTable."Key Code4" := '';
                  GenericTempTable."Key Code5" := PostedHourLine."Employee No.";
                  GenericTempTable."Key Code6" := JobType;
                  GenericTempTable.INSERT;
                END;
                GenericTempTable.Text1 := lvTempCompanyRec.Name;
                IF PostedHourLine."Receiving Company" <> '' THEN
                  GenericTempTable.Text2 := PostedHourLine."Receiving Company"
                ELSE
                  GenericTempTable.Text2 := lvTempCompanyRec.Name;
                IF PostedHourLine."Project No." <> '' THEN BEGIN
                  GenericTempTable.Text3 := Job.Description;
                  GenericTempTable.Text4 := Job.TABLECAPTION;
                END;
                IF PostedHourLine."Service Order No." <> '' THEN BEGIN
                  GenericTempTable.Text3 := ServiceOrder.Description;
                  IF CollectBaseOrder THEN
                    GenericTempTable.Text4 := ServiceOrder.FIELDCAPTION("Base Service Order No.")
                  ELSE
                    GenericTempTable.Text4 := ServiceOrder.TABLECAPTION;
                  IF PostedHourLine."Service Order No." <> JobNo THEN BEGIN
                    IF STRPOS(GenericTempTable.Text5, PostedHourLine."Service Order No.") = 0 THEN
                      IF GenericTempTable.Text5 = '' THEN
                        GenericTempTable.Text5 := PostedHourLine."Service Order No."
                      ELSE
                        GenericTempTable.Text5 := GenericTempTable.Text5 + '-' + PostedHourLine."Service Order No.";
                  END;
                END;
                GenericTempTable.Dec1 += PostedHourLine.Monday;
                GenericTempTable.Dec2 += PostedHourLine.Tuesday;
                GenericTempTable.Dec3 += PostedHourLine.Wednesday;
                GenericTempTable.Dec4 += PostedHourLine.Thursday;
                GenericTempTable.Dec5 += PostedHourLine.Friday;
                GenericTempTable.Dec6 += PostedHourLine.Saturday;
                GenericTempTable.Dec7 += PostedHourLine.Sunday;
                GenericTempTable.MODIFY;
              END;
            UNTIL PostedHourLine.NEXT = 0;
        UNTIL lvTempCompanyRec.NEXT = 0;
    END;

    BEGIN
    END.
