    PROCEDURE ProcessSelection@1210190000();
    VAR
      EstLine@1210190000 : Record 11012154;
      RcpLine@1210190002 : Record 11012156;
      LastLino@1210190001 : Integer;
    BEGIN
      LastLino :=0;
      EstLine.COPY("Estimate Line OLD");
      IF EstLine.FINDSET THEN BEGIN
        REPEAT
          IF EstLine.Recipe <> '' THEN BEGIN
            RcpLine.SETRANGE("Estimate No.", EstLine."Estimate No.");
            RcpLine.SETRANGE("Sub-Estimate No.", EstLine."Sub-Estimate No.");
            RcpLine.SETRANGE("Part Group", EstLine."Part Group");
            RcpLine.SETRANGE(Part, EstLine.Part);
            RcpLine.SETRANGE("Estimate Line", EstLine."Line No.");
            IF RcpLine.FINDSET THEN BEGIN
              REPEAT
                TempLine.INIT;
                TempLine."Estimate No." := RcpLine."Estimate No.";
                TempLine."Sub-Estimate No." := RcpLine."Sub-Estimate No.";
                TempLine."Part Group" := RcpLine."Part Group";
                TempLine.Part := RcpLine.Part;
                IF PartRec.GET(TempLine."Part Group", TempLine.Part) THEN BEGIN
                  TempLine.Chapter := PartRec.Chapter;
                  TempLine.Paragraph := PartRec.Paragraph;
                  TempLine.Level := PartRec.Level;
                END;
                TempLine."Estimate Line No." := RcpLine."Estimate Line";
                TempLine.Description := RcpLine.Description;
                IF RcpLine."Ignore Recipe Quantity" THEN  //db, 17-03-11: M26001
                 TempLine.Quantity := RcpLine.Quantity * EstLine."Part Quantity"
                ELSE
                  TempLine.Quantity := RcpLine.Quantity * EstLine."Part Quantity" * EstLine.Quantity;
                TempLine."Unit of Measure" := RcpLine."Unit of Measure";
                IF RcpLine.Hours <> 0 THEN BEGIN
                  LastLino := LastLino + 1;
                  TempLine."Line No." := LastLino;
                  TempLine."Cost Object" := RcpLine."Cost Object Labor";
                  TempLine.Hours := RcpLine."Cumulative Hours";  //db, 17-03-11: M26001
                  TempLine."Labor Amount" := RcpLine."Cumulative Labor";  //db, 17-03-11: M26001
                  TempLine.INSERT;
                END;
                IF RcpLine."Material Amount" <> 0 THEN BEGIN
                  LastLino := LastLino + 1;
                  TempLine."Line No." := LastLino;
                  TempLine."Cost Object" := RcpLine."Cost Object Material";
                  TempLine."Material Amount" := RcpLine."Cumulative Material";  //db, 17-03-11: M26001
                  TempLine.INSERT;
                END;
                IF RcpLine."Subcontracting Amount" <> 0 THEN BEGIN
                  LastLino := LastLino + 1;
                  TempLine."Line No." := LastLino;
                  TempLine."Cost Object" := RcpLine."Cost Object Subcontracting";
                  TempLine."Subcontracting Amount" := RcpLine."Cumulative Subcontracting";  //db, 17-03-11: M26001
                  TempLine.INSERT;
                END;
                IF RcpLine."Plant Amount" <> 0 THEN BEGIN
                  LastLino := LastLino + 1;
                  TempLine."Line No." := LastLino;
                  TempLine."Cost Object" := RcpLine."Cost Object Plant";
                  TempLine."Plant Amount" := RcpLine."Cumulative Plant";  //db, 17-03-11: M26001
                  TempLine.INSERT;
                END;
              UNTIL RcpLine.NEXT = 0;
            END;
          END ELSE BEGIN
            TempLine.INIT;
            TempLine."Estimate No." := EstLine."Estimate No.";
            TempLine."Sub-Estimate No." := EstLine."Sub-Estimate No.";
            TempLine."Part Group" := EstLine."Part Group";
            TempLine.Part := EstLine.Part;
            IF PartRec.GET(TempLine."Part Group", TempLine.Part) THEN BEGIN
              TempLine.Chapter := PartRec.Chapter;
              TempLine.Paragraph := PartRec.Paragraph;
              TempLine.Level := PartRec.Level;
            END;
            TempLine."Estimate Line No." := EstLine."Line No.";
            TempLine.Description := EstLine.Description;
            TempLine.Quantity := EstLine.Quantity * EstLine."Part Quantity";  //db, 17-03-11: M26001
            TempLine."Unit of Measure" := EstLine."Unit of Measure";
            IF EstLine.Hours <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Labor";
              TempLine.Hours := EstLine."Cumulative Hours";  //db, 17-03-11: M26001
              TempLine."Labor Amount" := EstLine."Cumulative Labor";  //db, 17-03-11: M26001
              TempLine.INSERT;
            END;
            IF EstLine."Material Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Material";
              TempLine."Material Amount" := EstLine."Cumulative Material";  //db, 17-03-11: M26001
              TempLine.INSERT;
            END;
            IF EstLine."Subcontracting Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Subcontracting";
              TempLine."Subcontracting Amount" := EstLine."Cumulative Subcontracting";  //db, 17-03-11: M26001
              TempLine.INSERT;
            END;
            IF EstLine."Plant Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Plant";
              TempLine."Plant Amount" := EstLine."Cumulative Plant";   //db, 17-03-11: M26001
              TempLine.INSERT;
            END;
          END;
        UNTIL EstLine.NEXT = 0;
      END;
    END;

    BEGIN
    END.
