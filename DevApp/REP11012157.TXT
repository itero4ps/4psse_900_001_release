    PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      EstimateLineNew@1100409000 : Record 11072072;
    BEGIN
      IF EstimateLineFilter.GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', EstimateLineNew.TABLECAPTION, "Estimate Line OLD".GETFILTERS));  //db, 05-09-14: C012769

    PROCEDURE GetEstimateData@1100485001();
    BEGIN
      Estimate.GET(EstimateLoop."No.");
      Estimate.CALCFIELDS("Customer Name");

      CLEAR(ModelCode);
      CLEAR(ModelQuan);
      EstimateModel.SETRANGE("Estimate No.", EstimateLoop."No.");
      IF EstimateModel.FINDSET THEN BEGIN
        counter := 0;
        REPEAT
          counter := counter + 1;
          IF counter < 100 THEN BEGIN
            ModelCode[counter] := EstimateModel.Model;
            ModelQuan[counter] := EstimateModel.Quantity;
          END;
        UNTIL EstimateModel.NEXT = 0;
      END ELSE BEGIN
        SubEstimate.SETRANGE("Estimate No.", EstimateLoop."No.");
        SubEstimate.SETFILTER("Source Estimate", '<>%1', '');
        IF SubEstimate.FINDFIRST THEN BEGIN
          EstimateModel.SETRANGE("Estimate No.", SubEstimate."Source Estimate");
          IF EstimateModel.FINDSET THEN BEGIN
            counter := 0;
            REPEAT
              counter := counter + 1;
              IF counter < 100 THEN BEGIN
                ModelCode[counter] := EstimateModel.Model;
                ModelQuan[counter] := EstimateModel.Quantity;
              END;
            UNTIL EstimateModel.NEXT = 0;
          END;
        END;
        SubEstimate.SETRANGE("Source Estimate");
      END;
    END;

    PROCEDURE GetSubEstimateData@1100485003();
    BEGIN
      SubEstimate.GET("Estimate Line OLD"."Estimate No.", "Estimate Line OLD"."Sub-Estimate No.");
      IF SubEstimate."Adjustment Factor Hours" = 1 THEN SubEstimate."Adjustment Factor Hours" := 0;
      IF SubEstimate."Adjustment Factor Material" = 1 THEN SubEstimate."Adjustment Factor Material" := 0;
      IF SubEstimate."Adjustment Factor Subcontr." = 1 THEN SubEstimate."Adjustment Factor Subcontr." := 0;
      IF SubEstimate."Adjustment Factor Plant" = 1 THEN SubEstimate."Adjustment Factor Plant" := 0;
    END;

    PROCEDURE GetEstimatePartData@1100485004();
    BEGIN
      SubEstimatePart.GET("Estimate Line OLD"."Estimate No.", "Estimate Line OLD"."Sub-Estimate No.",
        "Estimate Line OLD"."Part Group", "Estimate Line OLD".Part);
      IF SubEstimatePart."Adjustment Factor Hours" = 1 THEN SubEstimatePart."Adjustment Factor Hours" := 0;
      IF SubEstimatePart."Adjustment Factor Material" = 1 THEN SubEstimatePart."Adjustment Factor Material" := 0;
      IF SubEstimatePart."Adjustment Factor Subcontr." = 1 THEN SubEstimatePart."Adjustment Factor Subcontr." := 0;
      IF SubEstimatePart."Adjustment Factor Plant" = 1 THEN SubEstimatePart."Adjustment Factor Plant" := 0;
    END;

    PROCEDURE GetEstimateLineData@1100485002();
    BEGIN
      CLEAR(EstLineQuan);
      EstimateModelLineQuantity.SETRANGE("Estimate No.", "Estimate Line OLD"."Estimate No.");
      EstimateModelLineQuantity.SETRANGE("Sub-Estimate No.", "Estimate Line OLD"."Sub-Estimate No.");
      EstimateModelLineQuantity.SETRANGE("Part Group", "Estimate Line OLD"."Part Group");
      EstimateModelLineQuantity.SETRANGE(Part, "Estimate Line OLD".Part);
      EstimateModelLineQuantity.SETRANGE("Estimate Line", "Estimate Line OLD"."Line No.");
      IF EstimateModelLineQuantity.FINDSET THEN BEGIN
        REPEAT
          counter := 0;
          REPEAT
            counter := counter + 1;
            IF ModelCode[counter] = EstimateModelLineQuantity.Model THEN
              EstLineQuan[counter] := EstimateModelLineQuantity.Quantity;
          UNTIL (ModelCode[counter] = '') OR (counter = 99);
        UNTIL EstimateModelLineQuantity.NEXT = 0;
      END;
    END;

    PROCEDURE GetRecipeLineData@1100485000();
    BEGIN
      CLEAR(RcpLineQuan);

      counter := 0;
      REPEAT
        counter := counter + 1;
        IF "Recipe Line OLD"."Estimate Quantity" <> 0 THEN
          RcpLineQuan[counter] := EstLineQuan[counter] * "Recipe Line OLD".Quantity / "Recipe Line OLD"."Estimate Quantity";
      UNTIL (ModelCode[counter] = '') OR (counter = 99);
    END;

    END.
