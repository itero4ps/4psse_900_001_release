    PROCEDURE DetermineSummarySheet@1100485010(Estimate@1100485001 : Record 11012151;VAR SummerySheet@1100525000 : Record 11012172);

    PROCEDURE DetermineSurchEstLine@1100485000(VAR EstimateLine@1100485000 : Record 11012154;SummarySheet@1100485001 : Record 11012172);
    VAR
      HoursOld@1100485002 : Decimal;
      LaborAmountOld@1100485006 : Decimal;
    BEGIN
      //save old values
      HoursOld := EstimateLine.Hours;
      LaborAmountOld := EstimateLine."Labor Amount";

      //determine surcharges for each 'Cost Type' line.
      TempSummarySheetLine1.SETRANGE(Type, TempSummarySheetLine1.Type::"Cost Type");
      IF TempSummarySheetLine1.FINDSET THEN
        REPEAT
          //these are the lines in which the Estimate Line exist
          IF EstLineInFilters(EstimateLine, TempSummarySheetLine1) THEN BEGIN
            IF (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Labor) OR
               (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Total)
            THEN BEGIN
              TotSurchargeAmount := 0;
              TotSurchargeHours := 0;
              SummaryLineInFilter(TempSummarySheetLine1, 0, EstimateLine."Labor Amount", EstimateLine.Hours);
              EstimateLine."Labor Amount" += TotSurchargeAmount;
              EstimateLine.Hours += TotSurchargeHours;
            END;

            IF (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Material) OR
               (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Total)
            THEN BEGIN
              TotSurchargeAmount := 0;
              TotSurchargeHours := 0;
              SummaryLineInFilter(TempSummarySheetLine1, 0, EstimateLine."Material Amount", EstimateLine.Hours);
              EstimateLine."Material Amount" += TotSurchargeAmount;
            END;

            IF (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Subcontracting) OR
               (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Total)
            THEN BEGIN
              TotSurchargeAmount := 0;
              TotSurchargeHours := 0;
              SummaryLineInFilter(TempSummarySheetLine1, 0, EstimateLine."Subcontracting Amount", EstimateLine.Hours);
              EstimateLine."Subcontracting Amount" += TotSurchargeAmount;
            END;

            IF (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Plant) OR
               (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Total)
            THEN BEGIN
              TotSurchargeAmount := 0;
              TotSurchargeHours := 0;
              SummaryLineInFilter(TempSummarySheetLine1, 0, EstimateLine."Plant Amount", EstimateLine.Hours);
              EstimateLine."Plant Amount" += TotSurchargeAmount;
            END;
          END;
        UNTIL TempSummarySheetLine1.NEXT = 0;

      EstimateLine.Amount := EstimateLine."Labor Amount" + EstimateLine."Material Amount" + EstimateLine."Subcontracting Amount" + EstimateLine."Plant Amount";

      IF HoursOld <> EstimateLine.Hours THEN
        IF (EstimateLine.Quantity <> 0) AND (EstimateLine.Rate <> 0) THEN
          EstimateLine.Norm := EstimateLine.Hours / EstimateLine.Quantity;

      IF LaborAmountOld <> EstimateLine."Labor Amount" THEN
        IF EstimateLine.Hours <> 0 THEN
          EstimateLine.Rate := EstimateLine."Labor Amount" / EstimateLine.Hours;

      EstimateLine."Cumulative Hours" := EstimateLine.Hours * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Labor" := EstimateLine."Labor Amount" * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Material" := EstimateLine."Material Amount" * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Subcontracting" := EstimateLine."Subcontracting Amount" * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Plant" := EstimateLine."Plant Amount" * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Total" := EstimateLine.Amount * EstimateLine."Part Quantity";
    END;

    PROCEDURE EstLineInFilters@1100485002(EstimateLine@1100485000 : Record 11012154;SummarySheetLine@1100485001 : Record 11012171) : Boolean;
    VAR
      TempEstimateLine@1100485003 : TEMPORARY Record 11012154;
    BEGIN
      //Determine if Estimate Line fits in filter of the 'Cost Type' line.
      TempEstimateLine := EstimateLine;
      TempEstimateLine.INSERT;

      TempEstimateLine.SETRANGE("Estimate No.", SummarySheetLine."Estimate No.");
      TempEstimateLine.SETFILTER("Sub-Estimate No.", SummarySheetLine."Sub-Estimate Filter");
      //TempEstimateLine.SETFILTER("Part Group", SummarySheetLine."Part Group Filter");
      TempEstimateLine.SETFILTER(Part, SummarySheetLine."Part Filter");
      TempEstimateLine.SETFILTER("Line Status", SummarySheetLine."Line Status Filter");
      TempEstimateLine.SETFILTER("Rate Code", SummarySheetLine."Rate Filter");
      TempEstimateLine.SETFILTER("Material Price", SummarySheetLine."Material Price Filter");
      TempEstimateLine.SETFILTER(Code, SummarySheetLine."Code Filter");
      CASE SummarySheetLine."Cost Type" OF
        SummarySheetLine."Cost Type"::Labor,
        SummarySheetLine."Cost Type"::Total:
          TempEstimateLine.SETFILTER("Cost Object Labor", SummarySheetLine."Cost Object Filter");
        SummarySheetLine."Cost Type"::Material,
        SummarySheetLine."Cost Type"::Total:
          TempEstimateLine.SETFILTER("Cost Object Material", SummarySheetLine."Cost Object Filter");
        SummarySheetLine."Cost Type"::Subcontracting,
        SummarySheetLine."Cost Type"::Total:
          TempEstimateLine.SETFILTER("Cost Object Subcontracting", SummarySheetLine."Cost Object Filter");
        SummarySheetLine."Cost Type"::Plant,
        SummarySheetLine."Cost Type"::Total:
          TempEstimateLine.SETFILTER("Cost Object Plant", SummarySheetLine."Cost Object Filter");
      END;
      //TempEstimateLine.SETFILTER(Recipe, '%1', '');

      EXIT(TempEstimateLine.FINDFIRST);
    END;

    PROCEDURE SummaryLineInFilter@1100485001(SummarySheetLine@1100485000 : Record 11012171;Level@1100485003 : Integer;BaseAmount@1100485004 : Decimal;BaseHours@1100485006 : Decimal);
    VAR
      TempSummarySheetLine@1100485001 : TEMPORARY Record 11012171;
      SummarySheetLine2@1100485002 : Record 11012171;
      SurchargeAmount@1100485005 : Decimal;
      SurchargeHours@1100485007 : Decimal;
    BEGIN
      //calculates 1 level subcharge: called recursive!

      TempSummarySheetLine."Estimate No." := SummarySheetLine."Estimate No.";
      TempSummarySheetLine.Summary := SummarySheetLine.Summary;
      TempSummarySheetLine."Row No." := SummarySheetLine."Row No.";
      TempSummarySheetLine.INSERT;

      TempSummarySheetLine2.SETRANGE("Estimate No.", SummarySheetLine."Estimate No.");
      TempSummarySheetLine2.SETRANGE(Summary, SummarySheetLine.Summary);
      IF TempSummarySheetLine2.FINDSET THEN
        REPEAT
          IF TempSummarySheetLine2."Row No." <> SummarySheetLine."Row No." THEN BEGIN
            TempSummarySheetLine.SETRANGE("Estimate No.", TempSummarySheetLine2."Estimate No.");
            TempSummarySheetLine.SETRANGE(Summary, TempSummarySheetLine2.Summary);
            TempSummarySheetLine.SETFILTER("Row No.", TempSummarySheetLine2."Surcharge Over Row");
            IF TempSummarySheetLine.FINDFIRST THEN BEGIN
              IF TempSummarySheetLine2.Type = TempSummarySheetLine2.Type::"Surcharge Amount" THEN BEGIN
                SurchargeAmount := BaseAmount * TempSummarySheetLine2."Surcharge Percentage" / 100;
                TotSurchargeAmount += SurchargeAmount;
                SummarySheetLine2.COPY(TempSummarySheetLine2); //Save
                SummaryLineInFilter(TempSummarySheetLine2, Level + 1, SurchargeAmount, 0);
                TempSummarySheetLine2.COPY(SummarySheetLine2); //Retrieve
              END;
              IF TempSummarySheetLine2.Type = TempSummarySheetLine2.Type::"Surcharge Hours" THEN BEGIN
                SurchargeHours := BaseHours * TempSummarySheetLine2."Surcharge Percentage" / 100;
                TempSummarySheetLine2.CALCFIELDS("Cost Rate");
                SurchargeAmount := SurchargeHours * TempSummarySheetLine2."Cost Rate";
                TotSurchargeHours += SurchargeHours;
                TotSurchargeAmount += SurchargeAmount;
                SummarySheetLine2.COPY(TempSummarySheetLine2); //Save
                SummaryLineInFilter(TempSummarySheetLine2, Level + 1, SurchargeAmount, SurchargeHours);
                TempSummarySheetLine2.COPY(SummarySheetLine2); //Retrieve
              END;
            END;
          END;
        UNTIL TempSummarySheetLine2.NEXT = 0;
    END;

            IF (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Plant) OR
               (TempSummarySheetLine1."Cost Type" = TempSummarySheetLine1."Cost Type"::Total)
            THEN BEGIN
              TotSurchargeAmount := 0;
              TotSurchargeHours := 0;
              SummaryLineInFilter(TempSummarySheetLine1, 0, RecipeLine."Plant Amount", RecipeLine.Hours);
              RecipeLine."Plant Amount" += TotSurchargeAmount;
            END;
          END;
        UNTIL TempSummarySheetLine1.NEXT = 0;

      RecipeLine.Amount := RecipeLine."Labor Amount" + RecipeLine."Material Amount" + RecipeLine."Subcontracting Amount" + RecipeLine."Plant Amount";

      IF HoursOld <> RecipeLine.Hours THEN
        IF (RecipeLine.Quantity <> 0) AND (RecipeLine.Rate <> 0) THEN
          RecipeLine.Norm := RecipeLine.Hours / RecipeLine.Quantity;

      IF LaborAmountOld <> RecipeLine."Labor Amount" THEN
        IF RecipeLine.Hours <> 0 THEN
          RecipeLine.Rate := RecipeLine."Labor Amount" / RecipeLine.Hours;

      RecipeLine.CALCFIELDS("Estimate Part Quantity");
      RecipeLine."Cumulative Hours" := RecipeLine.Hours * RecipeLine."Estimate Part Quantity";
      RecipeLine."Cumulative Labor" := RecipeLine."Labor Amount" * RecipeLine."Estimate Part Quantity";
      RecipeLine."Cumulative Material" := RecipeLine."Material Amount" * RecipeLine."Estimate Part Quantity";
      RecipeLine."Cumulative Subcontracting" := RecipeLine."Subcontracting Amount" * RecipeLine."Estimate Part Quantity";
      RecipeLine."Cumulative Plant" := RecipeLine."Plant Amount" * RecipeLine."Estimate Part Quantity";
      //RecipeLine."Cumulative Total" := RecipeLine.Amount * RecipeLine."Estimate Part Quantity";
    END;

    PROCEDURE DetermineAmountRecipeEstLine@1100485017(VAR EstimateLine@1100485000 : Record 11012154);
    VAR
      RecipeLine@1100485001 : Record 11012156;
    BEGIN
      EstimateLine.Hours := 0;
      EstimateLine."Labor Amount" := 0;
      EstimateLine."Material Amount" := 0;
      EstimateLine."Subcontracting Amount" := 0;
      EstimateLine."Plant Amount" := 0;
      EstimateLine.Amount := 0;
      EstimateLine.Norm := 0;

      RecipeLine.SETRANGE("Estimate No.", EstimateLine."Estimate No.");

          EstimateLine.Amount += RecipeLine.Amount * EstimateLine.Quantity;

      IF EstimateLine.Quantity <> 0 THEN
        EstimateLine.Norm := EstimateLine.Hours / EstimateLine.Quantity;
      IF EstimateLine.Hours <> 0 THEN
        EstimateLine.Rate := EstimateLine."Labor Amount" / EstimateLine.Hours;

      EstimateLine."Cumulative Hours" := EstimateLine.Hours * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Labor" := EstimateLine."Labor Amount" * EstimateLine."Part Quantity";

      EstimateLine."Cumulative Plant" := EstimateLine."Plant Amount" * EstimateLine."Part Quantity";
      EstimateLine."Cumulative Total" := EstimateLine.Amount * EstimateLine."Part Quantity";
    END;

      SumLineRec.SETFILTER("Estimate No.", SumRec."Estimate No.");
      SumLineRec.SETFILTER(Summary, SumRec.Summary);

                IF TmpSubEstRec.INSERT THEN;
              UNTIL SubEstRec.NEXT = 0;
            END;

        EXIT(TRUE);
