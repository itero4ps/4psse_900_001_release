OBJECT Report 11012539 Transfer Option Diff to Budget

  OBJECT-PROPERTIES

    Date=;

  PROPERTIES
  {
    CaptionML=ENU=Transfer Option Difference to Budget;
    OnInitReport=BEGIN
                   GLSetup.GET; //mg.c, 27-05-11
                 END;

    OnPreReport=BEGIN
                  IF Transfer2Budget THEN BEGIN
                    IF DimValue= '' THEN
                      ERROR(Text001);
                    DimVal.GET(GLSetup."Global Dimension 2 Code", DimValue); //mg.c, 27-05-11
                    IF DimVal."Cost Type" = DimVal."Cost Type"::Revenue THEN
                      DimVal.FIELDERROR("Cost Type");
                    DimvalCosttype := DimVal."Cost Type";
                  END;
                  IF Job.GETFILTERS = '' THEN
                    ERROR(Text002);
                END;

  }
  DATASET
  {
    { 1100525002;;DataItem;                  ;
               DataItemTable=Table2000000026;
               DataItemTableView=SORTING(Number)

    { 1100525005;0;DataItem;                 ;

               OnPreDataItem=BEGIN
                               CurrReport.CREATETOTALS(Option."Contract Amount - Nett Amount");
                             END;

               ReqFilterFields=No. }

    { 1100525006;1;DataItem;                 ;
               DataItemTable=Table11012502;
               DataItemTableView=SORTING(Project No.,Option,Plot No.,Status,House Model,Main Group,Group,Sub Group)
                                 WHERE(Option=FILTER(<>''),
                                       Plot No.=FILTER(<>''),
                                       Status=FILTER(Order..Invoice));
               OnPreDataItem=BEGIN
                               CurrReport.CREATETOTALS("Contract Amount - Nett Amount");
                             END;

               OnAfterGetRecord=BEGIN
                                  Difference:= "Contract Amount - Nett Amount";
                                  IF Difference = 0 THEN
                                    CurrReport.SKIP;

                                  IF Transfer2Budget THEN BEGIN
                                    BudgetLineRec.SETRANGE("Project No.","Project No.");
                                    BudgetLineRec.SETRANGE("Plot No.","Plot No.");
                                    BudgetLineRec.SETRANGE("House Model","House Model");
                                    BudgetLineRec.SETRANGE("Main Group","Main Group");
                                    BudgetLineRec.SETRANGE(Group,Group);
                                    BudgetLineRec.SETRANGE("Sub Group","Sub Group");
                                    BudgetLineRec.SETRANGE(Option,Option);
                                    IF BudgetLineRec.FINDFIRST THEN BEGIN
                                      REPEAT
                                        VarLine:= BudgetLineRec."Line No." + 10000;    // make new line no out of highest used line no
                                        IF BudgetLineRec."Cost Object" = DimValue THEN
                                          BudgetLineRec.DELETE(TRUE);
                                      UNTIL BudgetLineRec.NEXT=0;
                                    END ELSE
                                      VarLine:=10000;

                                    BudgetLineRec.INIT;
                                    BudgetLineRec.VALIDATE("Project No.","Project No.");
                                    BudgetLineRec.VALIDATE("Plot No.","Plot No.");
                                    BudgetLineRec.VALIDATE("House Model","House Model");
                                    BudgetLineRec.VALIDATE("Main Group","Main Group");
                                    BudgetLineRec.VALIDATE(Group,Group);
                                    BudgetLineRec.VALIDATE("Sub Group","Sub Group");
                                    BudgetLineRec.VALIDATE(Option,Option);

                                    BudgetLineRec."Line No.":= VarLine;

                                    BudgetLineRec."Cost Type":=DimvalCosttype;
                                    BudgetLineRec.VALIDATE("Cost Type");
                                    BudgetLineRec."Cost Object":= DimValue;
                                    BudgetLineRec.VALIDATE("Cost Object");
                                    BudgetLineRec.Quantity:= 1;
                                    IF BudgetLineRec."Cost Type"= BudgetLineRec."Cost Type"::Labor THEN BEGIN
                                      BudgetLineRec.VALIDATE(Norm, 1);
                                      BudgetLineRec.VALIDATE(Rate, Difference);
                                    END ELSE
                                      BudgetLineRec.VALIDATE(Price, Difference);

                                    BudgetLineRec.Description := STRSUBSTNO(Text000, Option);
                                    BudgetLineRec.INSERT(TRUE);
                                  END;
                                END;

               ReqFilterFields=Plot No.,House Model,Main Group,Group,Sub Group,Option;
               DataItemLink=Project No.=FIELD(No.) }
