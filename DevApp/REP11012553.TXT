    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Plant Inventory".GETFILTERS <> '' THEN

    END;

    END;

    PROCEDURE SameRec@1() : Boolean;
    BEGIN
      IF (PlantInventory.Location <> "Plant Inventory".Location) OR
         (PlantInventory."Plant Type" <> "Plant Inventory"."Plant Type")
      THEN
        EXIT(FALSE);

      IF Details = Details::PlantNo THEN
        EXIT(PlantInventory."Plant No." = "Plant Inventory"."Plant No.");

      IF Details = Details::RateCode THEN
        EXIT((PlantInventory."Plant No."="Plant Inventory"."Plant No.") AND (PlantInventory."Rate Code"="Plant Inventory"."Rate Code"));

      EXIT(TRUE);
    END;

    {
      //**4PS.gb DPA Extra code to assign right filters on records without containing department code themselves
    }
