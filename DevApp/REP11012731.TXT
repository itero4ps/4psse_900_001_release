    PROCEDURE MarkServiceOrders@1210190000(IGlazingServiceCall@1210190001 : Record 11125777;VAR OServiceOrder@1210190000 : Record 11012823);
    BEGIN
      OServiceOrder.RESET;
      IF IGlazingServiceCall."Service Order No." <> '' THEN BEGIN //older type of glazing call
        OServiceOrder."No." := IGlazingServiceCall."Service Order No.";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 1 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 1 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 2 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 2 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 3 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 3 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 4 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 4 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 5 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 5 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      IF IGlazingServiceCall."Appointment 6 Service Order" <> '' THEN BEGIN
        OServiceOrder."No." := IGlazingServiceCall."Appointment 6 Service Order";
        OServiceOrder.MARK(TRUE);
      END;
      OServiceOrder.MARKEDONLY(TRUE);
    END;

    BEGIN
    END.
