    PROCEDURE IsValidServiceOrder@1100528609(ServiceOrder@1100528601 : Record 11012823;VAR ErrorMessage@1100528600 : Text[250]) : Boolean;
    VAR
      Customer@1100528602 : Record 18;
      CollectSOInvoicesBy@1100409000 : Option;
    BEGIN
      IF ServiceOrder."Customer No." = '' THEN
        ErrorMessage := STRSUBSTNO(Text001, ServiceOrder.FIELDCAPTION("Customer No."));

      IF NOT Customer.GET(ServiceOrder."Customer No.") THEN
        ErrorMessage := STRSUBSTNO(Text002, ServiceOrder.FIELDCAPTION("Customer No."), ServiceOrder."Customer No.");

      CollectSOInvoicesBy := Customer."Collect SO Invoices By";
      IF (CollectSOInvoicesBy = Customer."Collect SO Invoices By"::"Service Contract") AND
         (ServiceOrder."Service Contract No." = '')
      THEN
        CollectSOInvoicesBy := Customer."Collect SO Invoices By"::Customer;

      CASE CollectSOInvoicesBy OF
        Customer."Collect SO Invoices By"::"Service Contract":
          IF ServiceOrder."Source Type" = ServiceOrder."Source Type"::Contract THEN
            IF ServiceOrder."Service Contract No." = '' THEN
              ErrorMessage := STRSUBSTNO(Text001, ServiceOrder.FIELDCAPTION("Service Contract No."));
        Customer."Collect SO Invoices By"::"Service Location":
          IF ServiceOrder."Service Location No." = '' THEN
            ErrorMessage := STRSUBSTNO(Text001, ServiceOrder.FIELDCAPTION("Service Location No."));
        Customer."Collect SO Invoices By"::"Service Order":
          IF ServiceOrder."No." = '' THEN
            ErrorMessage := STRSUBSTNO(Text001, ServiceOrder.FIELDCAPTION("No."));
      END;

      IF ServiceOrder."Base Service Order No." <> ServiceOrder."No." THEN
        ErrorMessage := Text003
      ELSE
        IF NOT NextOrdersMatchPreliminaries(ServiceOrder."No.") THEN
          ErrorMessage := STRSUBSTNO(Text006, ServiceOrder."No.");

      EXIT(ErrorMessage = '');
    END;

    PROCEDURE IsValidBillToCustomer@1100528603(BillToCustomerNo@1100528601 : Code[20];VAR ErrorMessage@1100528600 : Text[250]) : Boolean;
    VAR
      Customer@1100528602 : Record 18;
      ServiceOrder@1100528603 : Record 11012823;
    BEGIN
      IF NOT Customer.GET(BillToCustomerNo) THEN
        ErrorMessage := STRSUBSTNO(Text002, ServiceOrder.FIELDCAPTION("Bill-to Customer No."), BillToCustomerNo);

      EXIT(ErrorMessage = '');
    END;

    PROCEDURE NextOrdersMatchPreliminaries@1100528600(IServiceOrderNo@1100528600 : Code[20]) : Boolean;
    VAR
      ServiceOrder@1100528601 : Record 11012823;
    BEGIN
      ServiceSetup.GET;
      ServiceOrder.SETCURRENTKEY("Base Service Order No.");
      ServiceOrder.SETRANGE("Base Service Order No.", IServiceOrderNo);
      ServiceOrder.SETRANGE("Base Service Order", FALSE);
      IF NOT OnlyReviewed AND NOT OnlyMadeChargeable THEN
        ServiceOrder.SETFILTER(Status, '<>%1&<>%2&<>%3',
          ServiceOrder.Status::"Technical Finished", ServiceOrder.Status::"Administrative Finished", ServiceOrder.Status::Finished);
      IF OnlyReviewed AND NOT OnlyMadeChargeable THEN
        ServiceOrder.SETFILTER("Actual Reference Point (Gantt)", '<>%1', ServiceSetup."Ref. Point SO Reviewed");
      IF NOT OnlyReviewed AND OnlyMadeChargeable THEN
        ServiceOrder.SETFILTER("Actual Reference Point (Gantt)", '<>%1', ServiceSetup."Ref. Point SO Make Chargeable");
      IF OnlyReviewed AND OnlyMadeChargeable THEN
        ServiceOrder.SETFILTER("Actual Reference Point (Gantt)", '<>%1&<>%2',
          ServiceSetup."Ref. Point SO Reviewed", ServiceSetup."Ref. Point SO Make Chargeable");
      EXIT(ServiceOrder.ISEMPTY);
    END;

    BEGIN
    END.
