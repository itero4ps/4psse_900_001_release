    LOCAL PROCEDURE GetReportName@1100528400() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));

    LOCAL PROCEDURE GetReportFilter@1100528401() : Text;
    BEGIN
      IF "G/L Entry".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO(
          '%1: %2%3%4: %5', "G/L Entry".TABLECAPTION, "G/L Entry".GETFILTERS,
          CrLf, Text002, WKRDimensionCode));
    END;

    LOCAL PROCEDURE GetWKRDimensionValue@1100528402(DimensionSetID@1100528401 : Integer;WKRDimensionCode@1100528400 : Code[20];VAR WKRDimensionValue@1100528402 : Record 349) : Boolean;
    VAR
      DimensionSetEntry@1100528403 : Record 480;
    BEGIN
      WKRDimensionValue.INIT;
      WKRDimensionValue."Dimension Code" := '';
      WKRDimensionValue.Code := '';

      IF DimensionSetID = 0 THEN
        EXIT(FALSE);

      IF DimensionSetEntry.GET(DimensionSetID, WKRDimensionCode) THEN BEGIN
        WKRDimensionValue.GET(WKRDimensionCode, DimensionSetEntry."Dimension Value Code");
        EXIT(TRUE);
      END ELSE
        EXIT(FALSE);
    END;

    LOCAL PROCEDURE CrLf@1100528403() CrLf : Text[2];
    BEGIN
      CrLf[1] := 13;
      CrLf[2] := 10;
    END;

    LOCAL PROCEDURE UpdateProgressBarOnWindow@1100528404(NewProgress@11012000 : Integer);
    BEGIN
      IF LastUpdateTime = 0T THEN
        LastUpdateTime := TIME;

      IF NoOfGLEntries > 0 THEN BEGIN
        NewProgress := ROUND(NewProgress / NoOfGLEntries * 100, 1);
        IF (NewProgress <> Progress) OR (TIME > LastUpdateTime + 5000) THEN BEGIN
          Window.UPDATE(1, 100 * NewProgress);
          LastUpdateTime := TIME;
        END;
        Progress := NewProgress;
      END;
    END;

    BEGIN
    END.
