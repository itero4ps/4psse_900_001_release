    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF ServiceContract.GETFILTERS <> '' THEN

    END;

    PROCEDURE CalcAmounts@1100528713(ServiceContractNo@1100528704 : Code[20];ServiceOrderNo@1100528700 : Code[20];SourceType@1100528703 : Option);
    VAR
      ServiceLedgerEntry@1100528701 : Record 11012819;
      ServiceOrder@1100528702 : Record 11012823;
    BEGIN
      TotalCostBeginPeriod := 0;
      TotalRevenueBeginPeriod := 0;
      TotalCostInPeriod := 0;
      TotalRevenueInPeriod := 0;
      ActualPreclosureCost := 0;
      ActualPreclosureRevenue := 0;
      ClosedCosts := 0;
      ClosedRevenues := 0;

      ServiceOrder.SETFILTER("Period Filter", PeriodFilter); // test period filter

      IF ServiceOrderNo = '' THEN
        ServiceLedgerEntry.SETCURRENTKEY("Service Contract No.","Source Type")
      ELSE
        ServiceLedgerEntry.SETCURRENTKEY("Service Order No.","Cost Plus Line No.");

      IF ServiceOrderNo = '' THEN
        ServiceLedgerEntry.SETRANGE("Service Contract No.", ServiceContractNo);
      ServiceLedgerEntry.SETRANGE("Service Order No.", ServiceOrderNo);
      ServiceLedgerEntry.SETRANGE("Source Type", SourceType);

      ServiceLedgerEntry.SETFILTER("Posting Date", '<%1', ServiceOrder.GETRANGEMIN("Period Filter"));
      IF ServiceLedgerEntry.FINDSET THEN
        REPEAT
          TotalCostBeginPeriod += ServiceLedgerEntry."Total Cost (LCY)";
          TotalRevenueBeginPeriod += ServiceLedgerEntry."Total Revenue (LCY)";
          IF ServiceOrderNo = '' THEN
            TotalRevenueBeginPeriod += -ServiceLedgerEntry."Closed Revenues";
        UNTIL ServiceLedgerEntry.NEXT = 0;

      ServiceLedgerEntry.SETRANGE("Posting Date", ServiceOrder.GETRANGEMIN("Period Filter"), ServiceOrder.GETRANGEMAX("Period Filter"));
      IF ServiceLedgerEntry.FINDSET THEN
        REPEAT
          TotalCostInPeriod += ServiceLedgerEntry."Total Cost (LCY)";
          TotalRevenueInPeriod += ServiceLedgerEntry."Total Revenue (LCY)";
          ActualPreclosureCost += ServiceLedgerEntry."Actual Preclosure Cost";
          ActualPreclosureRevenue += ServiceLedgerEntry."Actual Preclosure Revenue";
          IF ServiceOrderNo = '' THEN
            ActualPreclosureRevenue += ServiceLedgerEntry."Actual Preclosure Result";
          ClosedCosts += ServiceLedgerEntry."Closed Costs";
          ClosedRevenues += -ServiceLedgerEntry."Closed Revenues";
        UNTIL ServiceLedgerEntry.NEXT = 0;
    END;

    BEGIN
    END.
