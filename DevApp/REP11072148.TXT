    PROCEDURE CalculatePayedAmnt@1100409008(No@1100409000 : Code[20];Level@1100409006 : 'Contract,Order') : Decimal;
    VAR
      ServiceLedgerEntry@1100409001 : Record 11012819;
      GenericTempTable@1100409002 : TEMPORARY Record 11020579;
      CustLedgerEntry@1100409003 : Record 21;
      PercPayed@1100409004 : Decimal;
      Payed@1100409005 : Decimal;
    BEGIN
      GenericTempTable.DELETEALL;

      CASE Level OF
        Level::Order:
          BEGIN
            ServiceLedgerEntry.SETCURRENTKEY("Service Order No.","Source Type","Cost Type","Global Dimension 2 Code","Cost Component",
              "Additional Cost","Cost Present","Revenue Present","Posting Date");
            ServiceLedgerEntry.SETRANGE("Service Order No.", No);
          END;
        Level::Contract:
          BEGIN
            ServiceLedgerEntry.SETCURRENTKEY("Service Contract No.", "Posting Date");
            ServiceLedgerEntry.SETRANGE("Service Contract No.", No);
            ServiceLedgerEntry.SETRANGE("Service Order No.", '');
            ServiceLedgerEntry.SETRANGE("Posting Date", ServiceContractCtrlPeriod."Starting Date", ServiceContractCtrlPeriod."Ending Date");
          END;
      END;
      ServiceLedgerEntry.SETRANGE("Revenue Present", TRUE);
      ServiceLedgerEntry.SETFILTER("Document Type", '%1|%2', ServiceLedgerEntry."Document Type"::"Sales Invoice", ServiceLedgerEntry."Document Type"::"Sales Credit Memo");
      IF ServiceLedgerEntry.FINDSET THEN BEGIN
        REPEAT
          IF NOT GenericTempTable.GET(ServiceLedgerEntry."Document No.") THEN BEGIN
            GenericTempTable.INIT;
            GenericTempTable."Key Code1" := ServiceLedgerEntry."Document No.";
            GenericTempTable.Dec1 := ServiceLedgerEntry."Total Revenue (LCY)";
            GenericTempTable.INSERT;
          END ELSE BEGIN
            GenericTempTable.Dec1 += ServiceLedgerEntry."Total Revenue (LCY)";
            GenericTempTable.MODIFY;
          END;
        UNTIL ServiceLedgerEntry.NEXT = 0;
      END;

      IF GenericTempTable.FINDFIRST THEN BEGIN
        REPEAT
          CustLedgerEntry.SETCURRENTKEY("Document No.");
          CustLedgerEntry.SETRANGE("Document No.", GenericTempTable."Key Code1");
          CustLedgerEntry.SETFILTER("Document Type", '%1|%2', CustLedgerEntry."Document Type"::Invoice, CustLedgerEntry."Document Type"::"Credit Memo");
          IF CustLedgerEntry.FINDFIRST THEN BEGIN
            CustLedgerEntry.CALCFIELDS("Amount (LCY)", "Remaining Amt. (LCY)");
            IF CustLedgerEntry."Amount (LCY)" <> 0 THEN BEGIN
              PercPayed := (-1 * GenericTempTable.Dec1) / CustLedgerEntry."Amount (LCY)";
              Payed := Payed + ROUND((PercPayed * (CustLedgerEntry."Amount (LCY)" - CustLedgerEntry."Remaining Amt. (LCY)")), 0.01);
            END;
          END;
        UNTIL GenericTempTable.NEXT = 0;
      END;


      EXIT(Payed*-1);
    END;

    PROCEDURE CalculateContractMaintBudget@1100409010(lContractCde@1100409003 : Code[20]) MaintBudget : Decimal;
    VAR
      ContractObjectMaintenance@1100409000 : Record 11071692;
      IContractObject@1100409001 : Record 11071702;
      ServiceContractArrangement@1100409002 : Record 11071705;
      DateFilterEnd@1100409004 : Date;
      MaintenanceManagement@1100409006 : Codeunit 11012829;
      ServiceSurchargeManagement@1100409005 : Codeunit 11012834;
      DepartmentCode@1100409007 : Code[20];
      LaborAmount@1100409008 : Decimal;
      Hours@1100409009 : Decimal;
      Material@1100409010 : Decimal;
      Subcontracting@1100409011 : Decimal;
      Plant@1100409012 : Decimal;
      Sundry@1100409013 : Decimal;
      TotalCost@1100409014 : Decimal;
      SurChargeAmount@1100409015 : Decimal;
      ServiceOrder@1100409016 : Record 11012823;
      ServiceObject@1100409017 : Record 11071691;
    BEGIN
      MaintBudget := 0;
      IContractObject.SETRANGE("Service Contract No.",lContractCde);
      IContractObject.SETFILTER("Starting Date", '<=%1', ServiceContractCtrlPeriod."Ending Date");
      IF IContractObject.FINDSET THEN BEGIN
        REPEAT
          DateFilterEnd := ServiceContractCtrlPeriod."Ending Date";
          IF (IContractObject."Ending Date" > 0D) AND (IContractObject."Ending Date" <= DateFilterEnd) THEN
            DateFilterEnd := IContractObject."Ending Date";

          WITH ContractObjectMaintenance DO BEGIN
            SETRANGE("Service Contract No.", lContractCde);
            SETRANGE("Object No.", IContractObject."Service Object No.");
            IF FINDSET THEN BEGIN
              REPEAT
                UpdateServiceContractArr(DateFilterEnd, TRUE, FALSE);
              UNTIL NEXT = 0;
            END;
          END;

          WITH ServiceContractArrangement DO BEGIN
            SETRANGE("Service Contract No.", IContractObject."Service Contract No.");
            SETRANGE("Object No.", IContractObject."Service Object No.");
            SETRANGE("Theoretical Maintenance Date", ServiceContractCtrlPeriod."Starting Date", DateFilterEnd);
            SETRANGE(Cancelled, FALSE);
            IF FINDSET THEN BEGIN
              REPEAT
                LaborAmount := 0;
                Hours := 0;
                Material := 0;
                Subcontracting := 0;
                Sundry := 0;
                Plant := 0;
                TotalCost := 0;

                CALCFIELDS("Service Order No.");
                IF NOT ServiceOrder.GET("Service Order No.") THEN BEGIN
                  ServiceOrder.INIT;
                  ServiceObject.GET("Object No.");
                  DepartmentCode := ServiceObject.GetDepartmentCode("Service-Contract"."No.");
                  IF "Service-Contract"."Budget Based on" = "Service-Contract"."Budget Based on"::"Maintenance Scheme" THEN BEGIN
                    IContractObject.CALCFIELDS(Quantity);
                    MaintenanceManagement.CalcBudgetLines2(6, "Object No.", "Maintenance Scheme","Theoretical Maintenance Date", IContractObject.Quantity, DepartmentCode,
                      LaborAmount, Hours, Material, Subcontracting, Plant, Sundry, TotalCost);

                    SurChargeAmount := ServiceSurchargeManagement.CalculateSCSurchargeAmount(
                      IContractObject."Service Contract No.", ServiceOrder."Source Type"::Contract,
                      "Theoretical Maintenance Date",
                      LaborAmount, Material, Subcontracting, Plant, Sundry, TotalCost);
                    MaintBudget := MaintBudget + TotalCost + SurChargeAmount;
                  END;
                END;
