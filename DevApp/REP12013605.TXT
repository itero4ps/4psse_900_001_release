    PROCEDURE CheckUser@1100285001() : Boolean;
    VAR
      exUserComp@1100285000 : Record 12013641;
      TmpUserID@1100285004 : Code[100];
      TmpPos@1100285003 : Integer;
    BEGIN
      IF exUserComp.GET(COMPANYNAME,USERID) THEN
        EXIT(exUserComp."All-Company Access to Ex Docs");

      TmpPos := STRPOS(USERID,'\');
      IF TmpPos <> 0 THEN BEGIN
        TmpUserID := COPYSTR(USERID,TmpPos + 1);
        IF exUserComp.GET(COMPANYNAME,TmpUserID) THEN
          EXIT(exUserComp."All-Company Access to Ex Docs");
      END;
    END;

    BEGIN
    END.
