    PROCEDURE CalculateOutstanding@1000000000();
    VAR
      PurchLine@1000000000 : Record 39;
      CurrExchRate@1100285000 : Record 330;
      TempDate@1100285001 : Date;
    BEGIN
      PurchOrderAmount := 0;

      TempNextToApprove.RESET;
      TempNextToApprove.DELETEALL;

      PurchLine.RESET;
      PurchLine.SETRANGE("Document Type","Purchase Header"."Document Type");
      PurchLine.SETRANGE("Document No.","Purchase Header"."No.");
      IF PurchLine.FINDSET THEN BEGIN
        REPEAT
          IF PurchLine."VAT Calculation Type" <> PurchLine."VAT Calculation Type"::"Full VAT" THEN BEGIN
          PurchOrderAmount := PurchOrderAmount + PurchLine."Outstanding Amount" / (1 + PurchLine."VAT %" / 100);
          PurchOrderAmountLCY := PurchOrderAmount;

          BuildNextApprover(PurchLine);
          END;
        UNTIL PurchLine.NEXT = 0;

        IF "Purchase Header"."Currency Code" <> '' THEN BEGIN
          CASE CurrencyDate OF
            CurrencyDate::"Today's date": TempDate := WORKDATE;
            CurrencyDate::"Document's posting date": TempDate := "Purchase Header"."Posting Date";
          END;

          PurchOrderAmountLCY :=
            ROUND(
              CurrExchRate.ExchangeAmtFCYToLCY(0,"Purchase Header"."Job No.", // 4PS
                WORKDATE,"Purchase Header"."Currency Code",
                PurchOrderAmount,CurrExchRate.ExchangeRate(0,"Purchase Header"."Job No.",TempDate,"Purchase Header"."Currency Code",FALSE),FALSE)); // 4PS
        END;
      END;
    END;

    PROCEDURE BuildNextApprover@1000000001(PurchLine@1100285001 : Record 39);
    VAR
      DocLineApp@1100285000 : Record 12013610;
    BEGIN
      DocLineApp.RESET;
      DocLineApp.SETCURRENTKEY("Document No.");
      DocLineApp.SETRANGE("Document Type", PurchLine."Document Type");
      DocLineApp.SETRANGE("Document No.", PurchLine."Document No.");
      DocLineApp.SETRANGE("Line No.", PurchLine."Line No.");
      DocLineApp.SETRANGE(Flowstatus,DocLineApp.Flowstatus::Current);
      IF DocLineApp.FINDFIRST THEN BEGIN
        TempNextToApprove.SETRANGE(Approver, DocLineApp.Approver);
        IF NOT TempNextToApprove.FINDFIRST THEN BEGIN
          TempNextToApprove.Approver := DocLineApp.Approver;
          TempNextToApprove.ID := PurchLine."Line No.";
          TempNextToApprove.INSERT;
        END;
      END;
    END;

    BEGIN
    END.
