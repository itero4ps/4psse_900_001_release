    PROCEDURE fLookUpDate@1(p_dateStart@11012000 : Date) : Date;
    VAR
      frmCalendar@11012001 : Page 11012054;
      dateReturnValue@11012002 : Date;
    BEGIN
      //------------------------------------------------------------
      //What:   Opens up the calendar form to get a date
      //
      //Input:  <p_dateStart> the date which will be pre-selected
      //        when the form is open.
      //
      //Output: It returns the last date the user chooses unless
      //        he presses the Cancel button in which case it will
      //        return the start date <p_dateStart>
      //
      //Who:    Jos‚ Femen¡as Ca¤uelo. 17-Jun-98 / 17-Jun-98
      //------------------------------------------------------------

      IF FORMAT(p_dateStart)<>'' THEN
        fSetStartDate( p_dateStart )
      ELSE
        fSetStartDate( WORKDATE );

      //frmCalendar.RUNMODAL();
      frmCalendar.LOOKUPMODE := TRUE;
      IF frmCalendar.RUNMODAL = ACTION::LookupOK THEN
        dateReturnValue := fGetChosenDate
      ELSE
        dateReturnValue := p_dateStart;
      CLEAR(frmCalendar);
      EXIT( dateReturnValue );
    END;

    PROCEDURE fSetStartDate@17(p_dateStart@11012000 : Date);
    BEGIN
      //------------------------------------------------------------------------------------------------------------
      //WHAT:  Sets the three globals dateInitial, dateReturnValue, dateChosen to the value of the parameter

      //INPUT: <p_dateStart> The date we want to set the globals to

      //OUTPUT: Nothing
      //
      //WHO:    JFC 22.06.99
      //------------------------------------------------------------------------------------------------------------
      dateInitial     := p_dateStart;
      dateReturnValue := p_dateStart;
      dateChosen      := p_dateStart;
    END;

      //
      //OUTPUT: Date -> dateReturnValue; the value chosen by the user
