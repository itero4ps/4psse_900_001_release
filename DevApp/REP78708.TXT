                                    IF NOT ItemExists THEN
                                      ItemExists := PurLineRec."Trade Item" <> '';

                                  CrMemoTot += PurCrMemoRec.COUNT;

                                    Perc[5] := (CrMemoTot / InvoicesTot) * 100;

                                    IF PurRcptRec.FINDFIRST THEN
                                      IF PurRcptRec."Received (line)" <> PurRcptRec."Invoiced (line)" THEN

                                END;
