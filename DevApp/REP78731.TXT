    PROCEDURE PurchaseLines@1000000006(VAR JobCde@1000000000 : ARRAY [11] OF Text[50];VAR PurchLineRec@1000000001 : Record 39);
    VAR
      lvJobRec@1000000002 : Record 11072003;
      lvAdviceOrderLine@1000000004 : Record 11012055;
      lvPurchaseHeader@1000000003 : Record 38;
      lvProjectLocation@1000000005 : Record 11012011;
      lvNameAuthor@1000000006 : Text[50];
    BEGIN
      WITH PurchLineRec DO BEGIN
        CALCFIELDS("Vendor Name");
        IF lvJobRec.GET("Job No.") THEN
          lvJobRec.CALCFIELDS("Name Project Manager");
        lvAdviceOrderLine.RESET;
        lvAdviceOrderLine.SETRANGE("Project No.","Job No.");
        lvAdviceOrderLine.SETRANGE("Order Date","Posting Date");
        lvAdviceOrderLine.SETRANGE(lvAdviceOrderLine."Trade Item",PurchLineRec."Trade Item");
        IF lvAdviceOrderLine.FINDFIRST THEN
           lvNameAuthor := lvAdviceOrderLine."Created by";
        IF lvPurchaseHeader.GET(lvPurchaseHeader."Document Type"::Order,"Document No.") THEN
          IF lvPurchaseHeader."Ship To Location" <> '' THEN
            IF lvProjectLocation.GET("Job No.", lvPurchaseHeader."Ship To Location") THEN;
        FormatPurchaseLineCde(JobCde,"Document No.","Job No.",lvJobRec.Description,lvJobRec."Name Project Manager",
        "Vendor Name",FORMAT(Quantity),"Trade Item",Description,lvNameAuthor,STRSUBSTNO('%1 (%2)',
        lvPurchaseHeader."Ship To Location",lvProjectLocation.City),FORMAT(TODAY));
      END;
    END;
