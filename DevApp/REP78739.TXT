    { 1100525000;;DataItem;                  ;
               DataItemTable=Table27;
               DataItemTableView=SORTING(Vendor No.)
                                 ORDER(Descending);
               OnPreDataItem=BEGIN
                               IF GETFILTER("Date Filter") = '' THEN
                                 ToDate := WORKDATE
                               ELSE
                                 ToDate := GETRANGEMAX("Date Filter");

                               SETFILTER("Date Filter", '..%1', ToDate);
                               IF GETFILTER("Location Filter") = '' THEN
                                 ERROR(Text000, FIELDCAPTION("Location Filter"));

                               IF GETRANGEMIN("Location Filter") <> GETRANGEMAX("Location Filter") THEN
                                 ERROR(Text001, Location.TABLECAPTION, FIELDCAPTION("Location Filter"));

                               Location.GET(GETRANGEMIN("Location Filter"));
                               Window.OPEN(Text003);
                             END;

               OnAfterGetRecord=VAR
                                  Item2@1100525001 : Record 27;
                                  ItemLedgerEntry@1100525000 : Record 32;
                                BEGIN
                                  Window.UPDATE(1, "No.");

                                  IF SKU.GET(Location.Code, "No.") THEN BEGIN
                                    Item."Safety Stock Quantity" := SKU."Safety Stock Quantity";
                                    Item."Maximum Inventory" := SKU."Maximum Inventory";
                                  END;

                                  Item2.COPY(Item);
                                  Item2.SETFILTER("Date Filter", '..%1', ToDate);
                                  AvailabilityMgt.ExpectedQtyOnHand(Item2, TRUE, 0, QtyAvailable, ToDate);
                                  Item2.SETFILTER("Date Filter", '..%1', CALCDATE('<+1D>', ToDate));
                                  AvailabilityMgt.ExpectedQtyOnHand(Item2, TRUE, 0, QtyAvailableTomorrow, CALCDATE('<+1D>', ToDate));
                                  Item2.SETFILTER("Date Filter", '..%1', CALCDATE('<+2D>', ToDate));
                                  AvailabilityMgt.ExpectedQtyOnHand(Item2, TRUE, 0, QtyAvailableDayAfterTomorrow, CALCDATE('<+2D>', ToDate));
                                  Item2.SETFILTER("Date Filter", '..%1', CALCDATE('<+7D>', ToDate));
                                  AvailabilityMgt.ExpectedQtyOnHand(Item2, TRUE, 0, QtyAvailableNextWeek, CALCDATE('<+7D>', ToDate));

                                  CASE UnderstockedOnly OF
                                    TRUE:
                                      BEGIN
                                        CASE Item."Safety Stock Quantity" = 0 OF
                                          TRUE: OrderingAdvice := FALSE;
                                          FALSE: OrderingAdvice := (QtyAvailable - Item."Safety Stock Quantity" <= 0);
                                        END;
                                        IF NOT OrderingAdvice THEN
                                          CurrReport.SKIP;
                                      END;
                                    FALSE:
                                      BEGIN
                                        // Alleen artikelen op voorraad in dit magazijn,
                                        // of artikelen die ooit hier op voorraad zijn geweest
                                        ItemLedgerEntry.SETCURRENTKEY("Item No.", Positive, "Location Code");
                                        ItemLedgerEntry.SETRANGE("Item No.", "No.");
                                        ItemLedgerEntry.SETRANGE(Positive, TRUE);
                                        ItemLedgerEntry.SETRANGE("Location Code", Location.Code);
                                        IF ItemLedgerEntry.ISEMPTY THEN
                                          CurrReport.SKIP;
                                      END;
                                  END;
                                  IF NOT Vendor.GET(Item."Vendor No.") THEN
                                    Vendor.INIT;
                                  ShowItem3  := TRUE;
                                  IF Item."Safety Stock Quantity" <> 0 THEN
                                    IF UnderstockedOnly THEN
                                      ShowItem3  := OrderingAdvice;
                                END;

               OnPostDataItem=BEGIN
                                Window.CLOSE;
                              END;
