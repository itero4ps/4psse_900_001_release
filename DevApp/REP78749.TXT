    PROCEDURE FillJobInfo@1100525002();
    VAR
      PurchaseLine@1100525000 : Record 39;
    BEGIN
      //**4PS
      JobNoLabelTxt := '';
      JobNoTxt := '';

      IF "Purchase Header"."Job No." <> '' THEN BEGIN
        JobNoLabelTxt := "Purchase Header".FIELDCAPTION("Job No.");
        JobNoTxt := "Purchase Header"."Job No.";
      END ELSE IF "Purchase Header"."Service Order No." <> '' THEN BEGIN
        JobNoLabelTxt := "Purchase Header".FIELDCAPTION("Service Order No.");
        JobNoTxt := "Purchase Header"."Service Order No.";
      END ELSE BEGIN
        PurchaseLine.SETRANGE("Document Type", "Purchase Header"."Document Type");
        PurchaseLine.SETRANGE("Document No.", "Purchase Header"."No.");
        PurchaseLine.SETFILTER(Type, '<>%1', PurchaseLine.Type::" ");
        PurchaseLine.SETFILTER("Job No.", '<>%1', '');
        IF PurchaseLine.FINDFIRST THEN BEGIN
          JobNoLabelTxt := "Purchase Header".FIELDCAPTION("Job No.");
          JobNoTxt := PurchaseLine."Job No.";
        END ELSE BEGIN
          PurchaseLine.SETRANGE("Document Type", "Purchase Header"."Document Type");
          PurchaseLine.SETRANGE("Document No.", "Purchase Header"."No.");
          PurchaseLine.SETFILTER(Type, '<>%1', PurchaseLine.Type::" ");
          PurchaseLine.SETFILTER("Service Order No.", '<>%1', '');
          IF PurchaseLine.FINDFIRST THEN BEGIN
            JobNoLabelTxt := "Purchase Header".FIELDCAPTION("Service Order No.");
            JobNoTxt := PurchaseLine."Service Order No.";
          END;
        END;
      END;
    END;
