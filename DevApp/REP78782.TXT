    PROCEDURE CalculateSurcharge@1000000001(pDocumentNo@1000000000 : Code[20];pLineNo@1000000001 : Integer) : Decimal;
    VAR
      PurchOrderCtrlLine@1000000002 : Record 11020221;
      lvDec@1000000003 : Decimal;
    BEGIN
      WITH PurchOrderCtrlLine DO BEGIN
        SETRANGE("Order No.", pDocumentNo);
        SETRANGE("Order Line No.", pLineNo);
        IF FINDFIRST THEN
          REPEAT
            lvDec += ("Overhead Surcharge Firm" + "Overhead Surcharge Soft");
          UNTIL NEXT = 0;
      END;
      EXIT(lvDec);
    END;
