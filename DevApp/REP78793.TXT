    VAR
      lResult@1000000000 : Text[30];
    BEGIN
      Result := ChapterNo;

      IF ParagraphNo = '00' THEN
        EXIT;

      Result := STRSUBSTNO('%1.%2', Result, ParagraphNo);

      IF SectionNo = '00' THEN
        EXIT;

      Result := STRSUBSTNO('%1.%2', Result, SectionNo);

      IF SubsectionNo = '00' THEN
        EXIT;

      IF SubsectionNo > '00' THEN BEGIN
        Result := STRSUBSTNO('%1', SubsectionNo);
        lResult := Result;
        IF Result = TempResult THEN
          Result := '';
        TempResult := lResult;
      END;
      //Result := STRSUBSTNO('%1.%2', Result, SubsectionNo);
    END;

    BEGIN
    END.
