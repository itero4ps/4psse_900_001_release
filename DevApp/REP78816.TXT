        END;
        4: BEGIN   //Oorzaak
          ServiceText.SETFILTER("Table Field",'94');
        END;
      END;
      IF ServiceText.FINDSET THEN BEGIN
        REPEAT
          i := i + 1;
          Text := Text + ' ' + ServiceText.Comment;
