    PROCEDURE samenvoegen@1000000000() niet_afdrukken : Boolean;
    VAR
      PurchaseLineRec@1000000010 : Record 39;
    BEGIN
      //ctb mk 2010-03-19; f/unctie om sommaties te maken en een regel per artikelnummer af te drukken.
      //samenvoegen op basis van artikelnummer
      //eerst alle waardes op 0 zetten
      niet_afdrukken:= FALSE;//altijd standaard wel afdrukken.
      "Purchase Line".Quantity := 0;
      aantal_afgegeven := 0;
      "Purchase Line"."Qty. to Receive" := 0; //is ook nog af te geven aantal!?!?
      "Purchase Line"."Line Amount" := 0;
      bedrag_afgegeven := 0;
      bedrag_aftegeven := 0;
      QuantToBeDelivered := 0;
      TempQuant := 0;

      PurchaseLineRec.RESET;
      PurchaseLineRec.SETCURRENTKEY("Document Type","Document No.","Line No.");
      "Purchase Header".COPYFILTER("Expected Receipt Date",PurchaseLineRec."Expected Receipt Date");//ivm o.a. datumfilters
      PurchaseLineRec.SETRANGE(PurchaseLineRec."Document Type",PurchaseLineRec."Document Type"::"Blanket Order");
      PurchaseLineRec.SETRANGE("Document No.","Purchase Line"."Document No.");
      PurchaseLineRec.SETRANGE("No.","Purchase Line"."No.");// per artikelnummer samenvoegen!!
      IF PurchaseLineRec.FINDFIRST THEN BEGIN
        //als eerste regel dan alles hierop samenvoegen
        IF PurchaseLineRec."Line No." = "Purchase Line"."Line No." THEN BEGIN
          REPEAT
            PurchaseLineRec.CALCFIELDS("Quantity Released");
            TempQuant := TempQuant + PurchaseLineRec.Quantity;
            "Purchase Line"."Line Amount" := "Purchase Line"."Line Amount" + PurchaseLineRec."Line Amount";
            aantal_afgegeven := aantal_afgegeven + PurchaseLineRec."Quantity Released";
            bedrag_afgegeven := bedrag_afgegeven + (PurchaseLineRec."Quantity Released" * PurchaseLineRec."Direct Unit Cost");
            QuantToBeDelivered := QuantToBeDelivered  + PurchaseLineRec.GetQuanToBeDelivered;
            bedrag_aftegeven := bedrag_aftegeven + (PurchaseLineRec.GetQuanToBeDelivered * PurchaseLineRec."Direct Unit Cost");
            "Purchase Line"."Qty. to Receive" := PurchaseLineRec.GetQuanToBeDelivered;
          UNTIL PurchaseLineRec.NEXT = 0;

          //ook cumulatief bijhouden voor eindtotaal
          som_aantal := som_aantal + TempQuant;
          som_bedrag := som_bedrag + "Purchase Line"."Line Amount";
          som_aantal_afgegeven := som_aantal_afgegeven + aantal_afgegeven;
          som_bedrag_afgegeven := som_bedrag_afgegeven + bedrag_afgegeven;
          som_aantal_aftegeven :=som_aantal_aftegeven + QuantToBeDelivered;
          som_bedrag_aftegeven := som_bedrag_aftegeven + bedrag_aftegeven;
          //anders andere regels, niet afdrukken
        END ELSE niet_afdrukken := TRUE;
      END;
    END;

    BEGIN
    {

      ctb mk 2010-0319 kopie van 78801. functie gebruikt om alles samen te voegen per artikelnummer. Rapport verder
      zoveel mogelijk intact gelaten.
      ctb mk 20100326 diverse secties verwijderd tbv afdrukken.
      datum = printdatum ipv documentdatum
    }
    END.
