    PROCEDURE GetActualHours@1100525034(JobNo@1000000000 : Code[20];CostObj@1000000001 : Code[20];Elem@1000000006 : Code[20];ExtContr@1000000002 : Code[20];VAR OvePostHoursDec@1000000007 : Decimal) : Decimal;
    VAR
      ProjEntryRec@1100525000 : Record 11072005;
      RegHourLine@1000000003 : Record 11012039;
      PostHoursDec@1000000004 : Decimal;
      RegHoursDec@1000000005 : Decimal;
    BEGIN
      ProjEntryRec.SETCURRENTKEY("Job No.",Element);
      ProjEntryRec.SETRANGE("Job No.",JobNo);
      ProjEntryRec.SETRANGE("Extension Contract", ExtContr);
      ProjEntryRec.SETRANGE(Element,Elem);
      IF ProjEntryRec.FINDSET THEN BEGIN
        REPEAT
          OvePostHoursDec := OvePostHoursDec + ProjEntryRec.Quantity;
          IF ProjEntryRec."Global Dimension 2 Code" = CostObj THEN
            PostHoursDec := PostHoursDec + ProjEntryRec.Quantity;
        UNTIL ProjEntryRec.NEXT = 0;
      END;

      IF CalcRegistgratedHours THEN BEGIN
        RegHourLine.SETRANGE("Project No.",JobNo);
        RegHourLine.SETRANGE("Cost Object",CostObj);
        RegHourLine.SETRANGE("Extension Contract", ExtContr);
        RegHourLine.SETRANGE(Element,Elem);
        IF RegHourLine.FINDSET THEN BEGIN
          REPEAT
            RegHoursDec := RegHoursDec + RegHourLine."Total Line";
          UNTIL RegHourLine.NEXT = 0;
        END;
        RegHourLine.SETRANGE("Cost Object");
        IF RegHourLine.FINDSET THEN BEGIN
          REPEAT
            OvePostHoursDec := OvePostHoursDec + RegHourLine."Total Line";
          UNTIL RegHourLine.NEXT = 0;
        END;
      END;

      EXIT(PostHoursDec + RegHoursDec);
    END;
