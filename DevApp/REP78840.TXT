    PROCEDURE BerOpenstSaldo@1000000005(InvoiceNo@1000000000 : Code[20];Vendor@1000000001 : Code[20]) TempDec : Decimal;
    VAR
      VendLedgEntryRec@1000000002 : Record 25;
    BEGIN
      VendLedgEntryRec.SETRANGE("Document Type", VendLedgEntryRec."Document Type"::Invoice);
      VendLedgEntryRec.SETRANGE("Document No.", InvoiceNo);
      VendLedgEntryRec.SETRANGE("Vendor No.", Vendor);
      IF VendLedgEntryRec.FINDSET THEN BEGIN
        REPEAT
          VendLedgEntryRec.CALCFIELDS("Remaining Amount");
          TempDec += VendLedgEntryRec."Remaining Amount";
        UNTIL VendLedgEntryRec.NEXT = 0;
      END;
    END;
