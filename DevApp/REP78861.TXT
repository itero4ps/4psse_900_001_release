                  IF Pakbonfilter = '' THEN
                    ERROR(Text001);

                  IF AfleverFilter = '' THEN
                    ERROR(Text002);

                  IF AfleverlocatieFilter = '' THEN
                    ERROR(Text003);

                  CompanyInfo.GET;
                  CompanyInfo.CALCFIELDS(Picture);
                  ProjectManagerName := '';

                  IF JobRec.GET(Pakbonfilter) THEN BEGIN
                    JobRec.CALCFIELDS("Name Project Manager");
                    IF JobRec."Country/Region Code" <> '' THEN BEGIN
                      Landen.GET(JobRec."Country/Region Code");
                      CountryText := Landen.Name;
                    END;
                    IF JobRec."Project Manager" <> '' THEN BEGIN
                      EmployeeRec.GET(JobRec."Project Manager");
                      IF (EmployeeRec.Extension = '') AND (EmployeeRec."E-Mail" = '') THEN
                        ProjectManagerName := STRSUBSTNO('%1 %2',ProjectMan,JobRec."Name Project Manager")
                      ELSE
                        ProjectManagerName := STRSUBSTNO('%1 %2 (%3 / %4)',ProjectMan,JobRec."Name Project Manager",
                        EmployeeRec.Extension, EmployeeRec."Company E-Mail"); //EmployeeRec."E-Mail"//240610
                    END;
                    Projectopdrachtgever.GET(JobRec."No.",JobRec."Bill-to Customer No.");
                    YourOrderNo := Projectopdrachtgever."Principal Reference";
                    Projectopdrachtgever.CALCFIELDS("Contact Person Name");
                    IF Projectopdrachtgever."Contact Person Name" <> '' THEN BEGIN
                      JobAddress[1] := Projectopdrachtgever."Contact Person Name";
                    END;

                    IF AfleverFilter = 'Vestiging' THEN BEGIN
                      Vestiging.GET(AfleverlocatieFilter);
                      JobRec.Name := Vestiging.Name;
                      JobAddress[2] := Vestiging.Address;
                      JobAddress[3] := STRSUBSTNO('%1  %2',Vestiging."Post Code",UPPERCASE(Vestiging.City));
                      CityText := UPPERCASE(Vestiging.City);
                      IF Vestiging."Country/Region Code" <> '' THEN BEGIN
                        IF Vestiging."Country/Region Code" <> 'NL' THEN
                          CurrReport.LANGUAGE := 1033;
                        Landen.GET(Vestiging."Country/Region Code");
                        JobAddress[4] := Landen.Name;
                      END;
                    END;

                    IF AfleverFilter = 'Projectlocatie' THEN BEGIN
                      Projectlocatie.GET(JobRec."No.",AfleverlocatieFilter);
                      IF (Projectlocatie."Name 2" <> '') THEN BEGIN
                        JobNameTxt1 := Projectlocatie."Name 2";
                        JobNameTxt := Projectlocatie.Name;
                      END ELSE BEGIN
                        JobNameTxt := '';
                        JobNameTxt1 := Projectlocatie.Name;
                      END;

                      i := 1;
                      IF Projectlocatie.Contact <> '' THEN BEGIN
                        JobAddress[i] := Projectlocatie.Contact;
                        i += 1;
                      END;
                      JobAddress[i] := Projectlocatie.Address;
                      i += 1;
                      IF Projectlocatie."Address 2" <> '' THEN BEGIN
                        JobAddress[i] := Projectlocatie."Address 2";
                        i += 1;
                      END;
                      JobAddress[i] := STRSUBSTNO('%1  %2',Projectlocatie."Post Code",UPPERCASE(Projectlocatie.City));
                      i += 1;
                      CityText := UPPERCASE(Projectlocatie.City);
                      IF Projectlocatie."Country/Region Code" <> '' THEN BEGIN
                        IF Projectlocatie."Country/Region Code" <> 'NL' THEN
                          CurrReport.LANGUAGE := 1033;
                        Landen.GET(Projectlocatie."Country/Region Code");
                        IF CurrReport.LANGUAGE <> 1033 THEN
                          JobAddress[i] := 'NEDERLAND'
                        ELSE
                          JobAddress[i] := Landen.Name;
                      END;
                    END;
                    //bepalen verzendcondities
                    ProjLocRec := Projectlocatie;
                    CustomerRec.GET(JobRec."Bill-to Customer No.");
                    IF CustomerRec."Shipment Method Code" <>'' THEN
                  //  ShipMentMethodText := STRSUBSTNO('%1 -  %2 - %3',
                  //                        CustomerRec."Shipment Method Code",
                  //                        CityText,
                  //                        JobAddress[4]);

                  END;

    PROCEDURE ShipmentMethodDescription@1000000000() : Text[30];
    VAR
      Popup@1000000000 : Dialog;
      Text099@1000000001 : TextConst;
      Language@1000000005 : Record 8;
      CountryCode@1000000002 : Code[2];
      CityCode@1000000003 : Code[30];
      ShipmentMethodTranslation@1000000004 : Record 463;
    BEGIN
      // JvL - Op verzoek van Pieter Zomerman shipmentmethod uitgebreid
      Language.SETRANGE("Windows Language ID", CurrReport.LANGUAGE);
      IF Language.FINDSET THEN;
      IF NOT ShipmentMethodTranslation.GET(CustomerRec."Shipment Method Code", Language.Code) THEN
        ShipmentMethodTranslation.GET(CustomerRec."Shipment Method Code", '');

      IF STRPOS(ShipmentMethodTranslation.Description, '%5') <> 0 THEN BEGIN
        Popup.OPEN(STRSUBSTNO(Text099, ShipmentMethodTranslation."Shipment Method"));
        WHILE CityCode = '' DO
          Popup.INPUT(2, CityCode   );
        WHILE CountryCode = '' DO
          Popup.INPUT(3, CountryCode);
      END;

      EXIT(STRSUBSTNO( ShipmentMethodTranslation.Description           ,
                       ProjLocRec.City                                 ,
                       ProjLocRec."Country/Region Code"                ,
                      'AMERSFOORT'                                     ,
                      'NL'                                             ,
                       CityCode + ' - ' + CountryCode                  ));
    END;

    BEGIN
    {

      ShipmentMethod description changed

      Projectvoorraad toegevoegd
    }
