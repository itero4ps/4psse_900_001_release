    PROCEDURE ShowCard@1100528600();
    VAR
      ServiceContract@1100528600 : Record 11012812;
    BEGIN
      ServiceContract.GET("Service Contract No.");
      CASE ServiceContract."Budget Based on" OF
        ServiceContract."Budget Based on"::"Maintenance Scheme":
          PAGE.RUN(PAGE::"Service-Contract Control", Rec);
        ServiceContract."Budget Based on"::Contract:
          PAGE.RUN(PAGE::"SC Control (Category) Control", Rec);
      END;
    END;

    PROCEDURE GetFinancialResult@1100528601() : Decimal;
    BEGIN
      CALCFIELDS("Revenue (Total)", "Actual Cost (Total)");
      EXIT("Contract Revenue" + "Revenue (Total)" - "Actual Cost (Total)");
    END;

    PROCEDURE IsControlUnusedInPreclosure@1100528602(VAR OServiceContrPreclDetail@1100528600 : Record 11071767) : Boolean;
    BEGIN
      OServiceContrPreclDetail.SETCURRENTKEY("Service Contract No.");
      OServiceContrPreclDetail.SETRANGE("Service Contract No.", "Service Contract No.");
      OServiceContrPreclDetail.SETRANGE("SC Control Reporting Date", "Reporting Date");
      EXIT(NOT OServiceContrPreclDetail.FINDFIRST);
    END;

    LOCAL PROCEDURE TestControlUnusedInPreclosure@1100528603();
    VAR
      ServiceContrPreclDetail@1100528600 : Record 11071767;
    BEGIN
      IF NOT IsControlUnusedInPreclosure(ServiceContrPreclDetail) THEN
        ERROR(Text000, "Service Contract No.", "Reporting Date",
          ServiceContrPreclDetail."Reporting Date", ServiceContrPreclDetail."Seq. No.");
    END;

    END.
