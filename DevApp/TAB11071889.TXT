      SalesLine@1100285500 : Record 37;
      SalesInvoiceLine@1100285501 : Record 113;
      SalesCrMemoLine@1100285502 : Record 115;
    BEGIN
      CASE "Document Type" OF
        "Document Type"::"Pro Forma Invoice","Document Type"::"Pro Forma Credit Memo":
          BEGIN
            SalesLine.SETRANGE("Document Type", "Document Type");
            SalesLine.SETRANGE("Document No.", "Document No.");
            SalesLine.SETRANGE("Job No.", "Job No.");
            SalesLine.SETRANGE("Service Contract No.", "Service Contract No.");
            SalesLine.SETRANGE("Plant Location", "Plant Location");
            //SalesLine.SETRANGE("Cost Type Cost Plus Line","Cost Type Cost Plus Line");
            SalesLine.SETRANGE("Extension Contract","Extension Contract");
            SalesLine.CALCSUMS(SalesLine.Amount);
            EXIT(SalesLine.Amount);
          END;
        "Document Type"::"Posted Invoice":
          BEGIN
            //SalesInvoiceLine.SETRANGE("Document Type", "Document Type");
            SalesInvoiceLine.SETRANGE("Document No.", "Document No.");
            SalesInvoiceLine.SETRANGE("Job No.", "Job No.");
            SalesInvoiceLine.SETRANGE("Service Contract No.", "Service Contract No.");
            SalesInvoiceLine.SETRANGE("Plant Location", "Plant Location");
            //SalesInvoiceLine.SETRANGE("Cost Type Cost Plus Line","Cost Type Cost Plus Line");
            SalesInvoiceLine.SETRANGE("Extension Contract","Extension Contract");
            SalesInvoiceLine.CALCSUMS(SalesInvoiceLine.Amount);
            EXIT(SalesInvoiceLine.Amount);
          END;
        "Document Type"::"Posted Credit Memo":
          BEGIN
            SalesCrMemoLine.SETRANGE("Document No.", "Document No.");
            SalesCrMemoLine.SETRANGE("Job No.", "Job No.");
            SalesCrMemoLine.SETRANGE("Service Contract No.", "Service Contract No.");
            SalesCrMemoLine.SETRANGE("Plant Location", "Plant Location");
            SalesCrMemoLine.SETRANGE("Cost Type Cost Plus Line","Cost Type Cost Plus Line");
            SalesCrMemoLine.CALCSUMS(SalesCrMemoLine.Amount);
            EXIT(SalesCrMemoLine.Amount);
          END;
      END;
    END;
