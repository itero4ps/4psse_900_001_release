    PROCEDURE DetermineBackupFile@1210190002(OldFile@1210190011 : Text[250]) NewFile : Text[250];
    VAR
      JobSetup@1210190005 : Record 315;
      lvPos@1210190002 : Integer;
      lvPosDot@1210190001 : Integer;
      lvPosSlash@1210190012 : Integer;
    BEGIN
      //**4PS
      NewFile := OldFile;

      lvPos := STRLEN(OldFile);
      REPEAT
        IF lvPosDot = 0 THEN
          IF STRPOS(COPYSTR(OldFile,lvPos,1), '.') <> 0 THEN
            lvPosDot := lvPos;
        IF lvPosSlash = 0 THEN
          IF STRPOS(COPYSTR(OldFile,lvPos,1), '\') <> 0 THEN
            lvPosSlash := lvPos;
        lvPos := lvPos - 1;
      UNTIL (lvPos = 0) OR ((lvPosDot <> 0) AND (lvPosSlash <> 0));

      IF (lvPosDot <> 0) AND (lvPosSlash <> 0) THEN BEGIN
        JobSetup.GET;
        JobSetup.TESTFIELD("Pathname Salary Backup Files");
        NewFile := JobSetup."Pathname Salary Backup Files" +
                   COPYSTR(OldFile, lvPosSlash+1, lvPosDot-lvPosSlash-1) + ' ' +
                   FORMAT(TODAY,0,'<Year4><Month,2><Day,2>') + '(' +
                   FORMAT(TIME,0,'<Hours24,2><Minutes,2>') + ')' +
                   COPYSTR(OldFile, lvPosDot, STRLEN(OldFile)-lvPosDot+1);
      END;

      EXIT(NewFile);
    END;

    PROCEDURE CheckIfCodeIsAccordingToFormat@1100528401(Code@11012005 : Code[20];FormatCode@1100528401 : Code[20];ObjectCaption@1100528400 : Text[100]);
    VAR
      x1@1000000001 : Integer;
      x2@1000000002 : Integer;
      TempCode1@11012007 : Code[20];
      TempCode2@11012006 : Code[20];
    BEGIN
      //**4PS
      IF STRLEN(Code) <> STRLEN(FormatCode) THEN
        ERROR(Text11012010, ObjectCaption, Code);
      TempCode1 := Code;
      TempCode2 := FormatCode;
      REPEAT
        x1 := STRPOS(TempCode1, '.');
        x2 := STRPOS(TempCode2, '.');
        IF x1 <> x2 THEN
          ERROR(Text11012010, ObjectCaption, Code);
        IF x1 <> 0 THEN
          TempCode1 := DELSTR(TempCode1, x1, 1);
        IF x2 <> 0 THEN
          TempCode2 := DELSTR(TempCode2, x2, 1);
      UNTIL (x1 = 0) OR (x2 = 0);
    END;
